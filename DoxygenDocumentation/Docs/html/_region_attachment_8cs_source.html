95 public void UpdateRegion () {
96 float[] uvs = this.uvs;
97 if (region == null) {
98 uvs[BLX] = 0;
99 uvs[BLY] = 0;
100 uvs[ULX] = 0;
101 uvs[ULY] = 1;
102 uvs[URX] = 1;
103 uvs[URY] = 1;
104 uvs[BRX] = 1;
105 uvs[BRY] = 0;
106 return;
107 }
108
109 float width = Width, height = Height;
110 float localX2 = width / 2;
111 float localY2 = height / 2;
112 float localX = -localX2;
113 float localY = -localY2;
114 bool rotated = false;
115 if (region is AtlasRegion) {
116 AtlasRegion region = (AtlasRegion)this.region;
117 localX += region.offsetX / region.originalWidth * width;
118 localY += region.offsetY / region.originalHeight * height;
119 if (region.degrees == 90) {
120 rotated = true;
121 localX2 -= (region.originalWidth - region.offsetX - region.packedHeight) / region.originalWidth * width;
122 localY2 -= (region.originalHeight - region.offsetY - region.packedWidth) / region.originalHeight * height;
123 } else {
124 localX2 -= (region.originalWidth - region.offsetX - region.packedWidth) / region.originalWidth * width;
125 localY2 -= (region.originalHeight - region.offsetY - region.packedHeight) / region.originalHeight * height;
126 }
127 }
128 float scaleX = ScaleX, scaleY = ScaleY;
129 localX *= scaleX;
130 localY *= scaleY;
131 localX2 *= scaleX;
132 localY2 *= scaleY;
133 float r = Rotation * MathUtils.DegRad, cos = (float)Math.Cos(r), sin = (float)Math.Sin(r);
134 float x = X, y = Y;
135 float localXCos = localX * cos + x;
136 float localXSin = localX * sin;
137 float localYCos = localY * cos + y;
138 float localYSin = localY * sin;
139 float localX2Cos = localX2 * cos + x;
140 float localX2Sin = localX2 * sin;
141 float localY2Cos = localY2 * cos + y;
142 float localY2Sin = localY2 * sin;
143 float[] offset = this.offset;
144 offset[BLX] = localXCos - localYSin;
145 offset[BLY] = localYCos + localXSin;
146 offset[ULX] = localXCos - localY2Sin;
147 offset[ULY] = localY2Cos + localXSin;
148 offset[URX] = localX2Cos - localY2Sin;
149 offset[URY] = localY2Cos + localX2Sin;
150 offset[BRX] = localX2Cos - localYSin;
151 offset[BRY] = localYCos + localX2Sin;
152
153 if (rotated) {
154 uvs[BLX] = region.u2;
155 uvs[BLY] = region.v;
156 uvs[ULX] = region.u2;
157 uvs[ULY] = region.v2;
158 uvs[URX] = region.u;
159 uvs[URY] = region.v2;
160 uvs[BRX] = region.u;
161 uvs[BRY] = region.v;
162 } else {
163 uvs[BLX] = region.u2;
164 uvs[BLY] = region.v2;
165 uvs[ULX] = region.u;
166 uvs[ULY] = region.v2;
167 uvs[URX] = region.u;
168 uvs[URY] = region.v;
169 uvs[BRX] = region.u2;
170 uvs[BRY] = region.v;
171 }
172 }

181 public void ComputeWorldVertices (Slot slot, float[] worldVertices, int offset, int stride = 2) {
182 if (sequence != null) sequence.Apply(slot, this);
183
184 float[] vertexOffset = this.offset;
185 Bone bone = slot.Bone;
186 float bwx = bone.worldX, bwy = bone.worldY;
187 float a = bone.a, b = bone.b, c = bone.c, d = bone.d;
188 float offsetX, offsetY;
189
190 // Vertex order is different from RegionAttachment.java
191 offsetX = vertexOffset[BRX]; // 0
192 offsetY = vertexOffset[BRY]; // 1
193 worldVertices[offset] = offsetX * a + offsetY * b + bwx; // bl
194 worldVertices[offset + 1] = offsetX * c + offsetY * d + bwy;
195 offset += stride;
196
197 offsetX = vertexOffset[BLX]; // 2
198 offsetY = vertexOffset[BLY]; // 3
199 worldVertices[offset] = offsetX * a + offsetY * b + bwx; // ul
200 worldVertices[offset + 1] = offsetX * c + offsetY * d + bwy;
201 offset += stride;
202
203 offsetX = vertexOffset[ULX]; // 4
204 offsetY = vertexOffset[ULY]; // 5
205 worldVertices[offset] = offsetX * a + offsetY * b + bwx; // ur
206 worldVertices[offset + 1] = offsetX * c + offsetY * d + bwy;
207 offset += stride;
208
209 offsetX = vertexOffset[URX]; // 6
210 offsetY = vertexOffset[URY]; // 7
211 worldVertices[offset] = offsetX * a + offsetY * b + bwx; // br
212 worldVertices[offset + 1] = offsetX * c + offsetY * d + bwy;
213 //offset += stride;
214 }
