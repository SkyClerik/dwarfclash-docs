75 void LateUpdate()
76 {
77 _frameCounter++;
78
79 // Only update a subset of renderers if enough frames have passed
80 if (_frameCounter > _skipFrames)
81 {
82 _frameCounter = 0; // Reset frame counter
83
84 // --- 1. Update a subset of renderers (Time-Slicing) ---
85 if (_registeredRenderers.Count > 0)
86 {
87 for (int i = 0; i < _renderersToUpdatePerFrame; i++)
88 {
89 if (_rendererUpdateIndex >= _registeredRenderers.Count)
90 {
91 _rendererUpdateIndex = 0; // Wrap around
92 }
93
94 _registeredRenderers[_rendererUpdateIndex].UpdateParts();
95 _rendererUpdateIndex++;
96 }
97 }
98 }
99
100 // --- 2. Calculate average Z for each renderer (fast) & sort them ---
101 foreach (var renderer in _registeredRenderers)
102 {
103 renderer.CalculateAverageZ();
104 }
105 _registeredRenderers.Sort((a, b) => b.AverageZ.CompareTo(a.AverageZ));
106
107 // --- 3. Merge cached data from all renderers in the new sorted order ---
108 foreach(var list in _drawBatch.Values) list.Clear();
109
110 foreach (var renderer in _registeredRenderers)
111 {
112 var cachedParts = renderer.GetCachedParts();
113 foreach (var pair in cachedParts)
114 {
115 Texture texture = pair.Key;
116 List<CrowdRenderer.BodyPartData> parts = pair.Value;
117
118 if (!_drawBatch.ContainsKey(texture))
119 {
120 _drawBatch[texture] = new List<CrowdRenderer.BodyPartData>(parts.Count);
121 }
122 _drawBatch[texture].AddRange(parts);
123 }
124 }
125
126 // --- 4. Draw everything ---
127 DrawAllParts();
128 }

130 void DrawAllParts()
131 {
132 _bufferIndex = 0;
133 foreach (var pair in _drawBatch)
134 {
135 var texture = pair.Key;
136 var partsList = pair.Value;
137 if (partsList.Count == 0) continue;
138
139 if (_bufferIndex >= _partsBuffers.Length)
140 {
141 Debug.LogError("Not enough compute buffers! Increase buffer count or reduce texture variety.");
142 continue;
143 }
144
145 var currentBuffer = _partsBuffers[_bufferIndex];
146 currentBuffer.SetData(partsList, 0, 0, partsList.Count);
147
148 if (!_runtimeMaterials.ContainsKey(texture))
149 {
150 var newMat = new Material(_crowdShader);
151 newMat.SetTexture("_MainTex", texture);
152 _runtimeMaterials[texture] = newMat;
153 }
154 var material = _runtimeMaterials[texture];
155
156 _mpb.SetBuffer("_Parts", currentBuffer);
157 Graphics.DrawMeshInstancedProcedural(
158 _quadMesh, 0, material,
159 new Bounds(Vector3.zero, new Vector3(1000f, 1000f, 1000f)),
160 partsList.Count, _mpb);
161
162 _bufferIndex++;
163 }
164 }
