38 [CustomEditor(typeof(BoneFollowerGraphic)), CanEditMultipleObjects]
39 public class BoneFollowerGraphicInspector : Editor {

46 #region Context Menu Item
47 [MenuItem("CONTEXT/SkeletonGraphic/Add BoneFollower GameObject")]
48 static void AddBoneFollowerGameObject (MenuCommand cmd) {
49 SkeletonGraphic skeletonGraphic = cmd.context as SkeletonGraphic;
50 GameObject go = EditorInstantiation.NewGameObject("BoneFollower", true, typeof(RectTransform));
51 Transform t = go.transform;
52 t.SetParent(skeletonGraphic.transform);
53 t.localPosition = Vector3.zero;
54
55 BoneFollowerGraphic f = go.AddComponent<BoneFollowerGraphic>();
56 f.skeletonGraphic = skeletonGraphic;
57 f.SetBone(skeletonGraphic.Skeleton.RootBone.Data.Name);
58
59 EditorGUIUtility.PingObject(t);
60
61 Undo.RegisterCreatedObjectUndo(go, "Add BoneFollowerGraphic");
62 }
63
64 // Validate
65 [MenuItem("CONTEXT/SkeletonGraphic/Add BoneFollower GameObject", true)]
66 static bool ValidateAddBoneFollowerGameObject (MenuCommand cmd) {
67 SkeletonGraphic skeletonGraphic = cmd.context as SkeletonGraphic;
68 return skeletonGraphic.IsValid;
69 }

72 void OnEnable () {
73 skeletonGraphic = serializedObject.FindProperty("skeletonGraphic");
74 boneName = serializedObject.FindProperty("boneName");
75 followBoneRotation = serializedObject.FindProperty("followBoneRotation");
76 followXYPosition = serializedObject.FindProperty("followXYPosition");
77 followZPosition = serializedObject.FindProperty("followZPosition");
78 followAttachmentZSpacing = serializedObject.FindProperty("followAttachmentZSpacing");
79 followLocalScale = serializedObject.FindProperty("followLocalScale");
80 followParentWorldScale = serializedObject.FindProperty("followParentWorldScale");
81 followSkeletonFlip = serializedObject.FindProperty("followSkeletonFlip");
82 maintainedAxisOrientation = serializedObject.FindProperty("maintainedAxisOrientation");
83
84 targetBoneFollower = (BoneFollowerGraphic)target;
85 if (targetBoneFollower.SkeletonGraphic != null)
86 targetBoneFollower.SkeletonGraphic.Initialize(false);
87
88 if (!targetBoneFollower.valid || needsReset) {
89 targetBoneFollower.Initialize();
90 targetBoneFollower.LateUpdate();
91 needsReset = false;
92 SceneView.RepaintAll();
93 }
94 }

96 public void OnSceneGUI () {
97 BoneFollowerGraphic tbf = target as BoneFollowerGraphic;
98 SkeletonGraphic skeletonGraphicComponent = tbf.SkeletonGraphic;
99 if (skeletonGraphicComponent == null) return;
100
101 Transform transform = skeletonGraphicComponent.transform;
102 Skeleton skeleton = skeletonGraphicComponent.Skeleton;
103 float positionScale = skeletonGraphicComponent.MeshScale;
104 Vector2 positionOffset = skeletonGraphicComponent.GetScaledPivotOffset();
105
106 if (string.IsNullOrEmpty(boneName.stringValue)) {
107 SpineHandles.DrawBones(transform, skeleton, positionScale, positionOffset);
108 SpineHandles.DrawBoneNames(transform, skeleton, positionScale, positionOffset);
109 Handles.Label(tbf.transform.position, "No bone selected", EditorStyles.helpBox);
110 } else {
111 Bone targetBone = tbf.bone;
112 if (targetBone == null) return;
113
114 SpineHandles.DrawBoneWireframe(transform, targetBone, SpineHandles.TransformContraintColor, positionScale, positionOffset);
115 Handles.Label(targetBone.GetWorldPosition(transform, positionScale, positionOffset),
116 targetBone.Data.Name, SpineHandles.BoneNameStyle);
117 }
118 }

120 override public void OnInspectorGUI () {
121 if (serializedObject.isEditingMultipleObjects) {
122 if (needsReset) {
123 needsReset = false;
124 foreach (Object o in targets) {
125 BoneFollower bf = (BoneFollower)o;
126 bf.Initialize();
127 bf.LateUpdate();
128 }
129 SceneView.RepaintAll();
130 }
131
132 EditorGUI.BeginChangeCheck();
133 DrawDefaultInspector();
134 needsReset |= EditorGUI.EndChangeCheck();
135 return;
136 }
137
138 if (needsReset && Event.current.type == EventType.Layout) {
139 targetBoneFollower.Initialize();
140 targetBoneFollower.LateUpdate();
141 needsReset = false;
142 SceneView.RepaintAll();
143 }
144 serializedObject.Update();
145
146 // Find Renderer
147 if (skeletonGraphic.objectReferenceValue == null) {
148 SkeletonGraphic parentRenderer = targetBoneFollower.GetComponentInParent<SkeletonGraphic>();
149 if (parentRenderer != null && parentRenderer.gameObject != targetBoneFollower.gameObject) {
150 skeletonGraphic.objectReferenceValue = parentRenderer;
151 Debug.Log("Inspector automatically assigned BoneFollowerGraphic.SkeletonGraphic");
152 }
153 }
154
155 EditorGUILayout.PropertyField(skeletonGraphic);
156 SkeletonGraphic skeletonGraphicComponent = skeletonGraphic.objectReferenceValue as SkeletonGraphic;
157 if (skeletonGraphicComponent != null) {
158 if (skeletonGraphicComponent.gameObject == targetBoneFollower.gameObject) {
159 skeletonGraphic.objectReferenceValue = null;
160 EditorUtility.DisplayDialog("Invalid assignment.", "BoneFollowerGraphic can only follow a skeleton on a separate GameObject.\n\nCreate a new GameObject for your BoneFollower, or choose a SkeletonGraphic from a different GameObject.", "Ok");
161 }
162 }
163
164 if (!targetBoneFollower.valid) {
165 needsReset = true;
166 }
167
168 if (targetBoneFollower.valid) {
169 EditorGUI.BeginChangeCheck();
170 EditorGUILayout.PropertyField(boneName);
171 needsReset |= EditorGUI.EndChangeCheck();
172
173 EditorGUILayout.PropertyField(followBoneRotation);
174 EditorGUILayout.PropertyField(followXYPosition);
175 EditorGUILayout.PropertyField(followZPosition);
176 if (followZPosition.boolValue == true) {
177 using (new SpineInspectorUtility.IndentScope())
178 EditorGUILayout.PropertyField(followAttachmentZSpacing, new GUIContent("Attachment Z Spacing"));
179 }
180 EditorGUILayout.PropertyField(followLocalScale);
181 EditorGUILayout.PropertyField(followParentWorldScale);
182 EditorGUILayout.PropertyField(followSkeletonFlip);
183 if ((followSkeletonFlip.hasMultipleDifferentValues || followSkeletonFlip.boolValue == false) &&
184 (followBoneRotation.hasMultipleDifferentValues || followBoneRotation.boolValue == true)) {
185 using (new SpineInspectorUtility.IndentScope())
186 EditorGUILayout.PropertyField(maintainedAxisOrientation);
187 }
188
189 //BoneFollowerInspector.RecommendRigidbodyButton(targetBoneFollower);
190 } else {
191 SkeletonGraphic boneFollowerSkeletonGraphic = targetBoneFollower.skeletonGraphic;
192 if (boneFollowerSkeletonGraphic == null) {
193 EditorGUILayout.HelpBox("SkeletonGraphic is unassigned. Please assign a SkeletonRenderer (SkeletonAnimation or SkeletonMecanim).", MessageType.Warning);
194 } else {
195 boneFollowerSkeletonGraphic.Initialize(false);
196
197 if (boneFollowerSkeletonGraphic.skeletonDataAsset == null)
198 EditorGUILayout.HelpBox("Assigned SkeletonGraphic does not have SkeletonData assigned to it.", MessageType.Warning);
199
200 if (!boneFollowerSkeletonGraphic.IsValid)
201 EditorGUILayout.HelpBox("Assigned SkeletonGraphic is invalid. Check target SkeletonGraphic, its SkeletonData asset or the console for other errors.", MessageType.Warning);
202 }
203 }
204
205 Event current = Event.current;
206 bool wasUndo = (current.type == EventType.ValidateCommand && current.commandName == "UndoRedoPerformed");
207 if (wasUndo)
208 targetBoneFollower.Initialize();
209
210 serializedObject.ApplyModifiedProperties();
211 }
212
213 }
