113 public Atlas (TextReader reader, string imagesDir, TextureLoader textureLoader) {
114 if (reader == null) throw new ArgumentNullException("reader", "reader cannot be null.");
115 if (imagesDir == null) throw new ArgumentNullException("imagesDir", "imagesDir cannot be null.");
116 if (textureLoader == null) throw new ArgumentNullException("textureLoader", "textureLoader cannot be null.");
117 this.textureLoader = textureLoader;
118
119 string[] entry = new string[5];
120 AtlasPage page = null;
121 AtlasRegion region = null;
122
123 Dictionary<string, Action> pageFields = new Dictionary<string, Action>(5);
124 pageFields.Add("size", () => {
125 page.width = int.Parse(entry[1], CultureInfo.InvariantCulture);
126 page.height = int.Parse(entry[2], CultureInfo.InvariantCulture);
127 });
128 pageFields.Add("format", () => {
129 page.format = (Format)Enum.Parse(typeof(Format), entry[1], false);
130 });
131 pageFields.Add("filter", () => {
132 page.minFilter = (TextureFilter)Enum.Parse(typeof(TextureFilter), entry[1], false);
133 page.magFilter = (TextureFilter)Enum.Parse(typeof(TextureFilter), entry[2], false);
134 });
135 pageFields.Add("repeat", () => {
136 if (entry[1].IndexOf('x') != -1) page.uWrap = TextureWrap.Repeat;
137 if (entry[1].IndexOf('y') != -1) page.vWrap = TextureWrap.Repeat;
138 });
139 pageFields.Add("pma", () => {
140 page.pma = entry[1] == "true";
141 });
142
143 Dictionary<string, Action> regionFields = new Dictionary<string, Action>(8);
144 regionFields.Add("xy", () => { // Deprecated, use bounds.
145 region.x = int.Parse(entry[1], CultureInfo.InvariantCulture);
146 region.y = int.Parse(entry[2], CultureInfo.InvariantCulture);
147 });
148 regionFields.Add("size", () => { // Deprecated, use bounds.
149 region.width = int.Parse(entry[1], CultureInfo.InvariantCulture);
150 region.height = int.Parse(entry[2], CultureInfo.InvariantCulture);
151 });
152 regionFields.Add("bounds", () => {
153 region.x = int.Parse(entry[1], CultureInfo.InvariantCulture);
154 region.y = int.Parse(entry[2], CultureInfo.InvariantCulture);
155 region.width = int.Parse(entry[3], CultureInfo.InvariantCulture);
156 region.height = int.Parse(entry[4], CultureInfo.InvariantCulture);
157 });
158 regionFields.Add("offset", () => { // Deprecated, use offsets.
159 region.offsetX = int.Parse(entry[1], CultureInfo.InvariantCulture);
160 region.offsetY = int.Parse(entry[2], CultureInfo.InvariantCulture);
161 });
162 regionFields.Add("orig", () => { // Deprecated, use offsets.
163 region.originalWidth = int.Parse(entry[1], CultureInfo.InvariantCulture);
164 region.originalHeight = int.Parse(entry[2], CultureInfo.InvariantCulture);
165 });
166 regionFields.Add("offsets", () => {
167 region.offsetX = int.Parse(entry[1], CultureInfo.InvariantCulture);
168 region.offsetY = int.Parse(entry[2], CultureInfo.InvariantCulture);
169 region.originalWidth = int.Parse(entry[3], CultureInfo.InvariantCulture);
170 region.originalHeight = int.Parse(entry[4], CultureInfo.InvariantCulture);
171 });
172 regionFields.Add("rotate", () => {
173 string value = entry[1];
174 if (value == "true")
175 region.degrees = 90;
176 else if (value != "false")
177 region.degrees = int.Parse(value, CultureInfo.InvariantCulture);
178 });
179 regionFields.Add("index", () => {
180 region.index = int.Parse(entry[1], CultureInfo.InvariantCulture);
181 });
182
183 string line = reader.ReadLine();
184 // Ignore empty lines before first entry.
185 while (line != null && line.Trim().Length == 0)
186 line = reader.ReadLine();
187 // Header entries.
188 while (true) {
189 if (line == null || line.Trim().Length == 0) break;
190 if (ReadEntry(entry, line) == 0) break; // Silently ignore all header fields.
191 line = reader.ReadLine();
192 }
193 // Page and region entries.
194 List<string> names = null;
195 List<int[]> values = null;
196 while (true) {
197 if (line == null) break;
198 if (line.Trim().Length == 0) {
199 page = null;
200 line = reader.ReadLine();
201 } else if (page == null) {
202 page = new AtlasPage();
203 page.name = line.Trim();
204 while (true) {
205 if (ReadEntry(entry, line = reader.ReadLine()) == 0) break;
206 Action field;
207 if (pageFields.TryGetValue(entry[0], out field)) field(); // Silently ignore unknown page fields.
208 }
209 textureLoader.Load(page, Path.Combine(imagesDir, page.name));
210 pages.Add(page);
211 } else {
212 region = new AtlasRegion();
213 region.page = page;
214 region.name = line;
215 while (true) {
216 int count = ReadEntry(entry, line = reader.ReadLine());
217 if (count == 0) break;
218 Action field;
219 if (regionFields.TryGetValue(entry[0], out field))
220 field();
221 else {
222 if (names == null) {
223 names = new List<string>(8);
224 values = new List<int[]>(8);
225 }
226 names.Add(entry[0]);
227 int[] entryValues = new int[count];
228 for (int i = 0; i < count; i++)
229 int.TryParse(entry[i + 1], NumberStyles.Any, CultureInfo.InvariantCulture, out entryValues[i]); // Silently ignore non-integer values.
230 values.Add(entryValues);
231 }
232 }
233 if (region.originalWidth == 0 && region.originalHeight == 0) {
234 region.originalWidth = region.width;
235 region.originalHeight = region.height;
236 }
237 if (names != null && names.Count > 0) {
238 region.names = names.ToArray();
239 region.values = values.ToArray();
240 names.Clear();
241 values.Clear();
242 }
243 region.u = region.x / (float)page.width;
244 region.v = region.y / (float)page.height;
245 if (region.degrees == 90) {
246 region.u2 = (region.x + region.height) / (float)page.width;
247 region.v2 = (region.y + region.width) / (float)page.height;
248
249 int tempSwap = region.packedWidth;
250 region.packedWidth = region.packedHeight;
251 region.packedHeight = tempSwap;
252 } else {
253 region.u2 = (region.x + region.width) / (float)page.width;
254 region.v2 = (region.y + region.height) / (float)page.height;
255 }
256 regions.Add(region);
257 }
258 }
259 }

261 static private int ReadEntry (string[] entry, string line) {
262 if (line == null) return 0;
263 line = line.Trim();
264 if (line.Length == 0) return 0;
265 int colon = line.IndexOf(':');
266 if (colon == -1) return 0;
267 entry[0] = line.Substring(0, colon).Trim();
268 for (int i = 1, lastMatch = colon + 1; ; i++) {
269 int comma = line.IndexOf(',', lastMatch);
270 if (comma == -1) {
271 entry[i] = line.Substring(lastMatch).Trim();
272 return i;
273 }
274 entry[i] = line.Substring(lastMatch, comma - lastMatch).Trim();
275 lastMatch = comma + 1;
276 if (i == 4) return 4;
277 }
