72 public static RegionAttachment ToRegionAttachment (this AtlasRegion region, string attachmentName, float scale = 0.01f, float rotation = 0f) {
73 if (string.IsNullOrEmpty(attachmentName)) throw new System.ArgumentException("attachmentName can't be null or empty.", "attachmentName");
74 if (region == null) throw new System.ArgumentNullException("region");
75
76 // (AtlasAttachmentLoader.cs)
77 RegionAttachment attachment = new RegionAttachment(attachmentName);
78
79 attachment.Region = region;
80 attachment.Path = region.name;
81 attachment.ScaleX = 1;
82 attachment.ScaleY = 1;
83 attachment.Rotation = rotation;
84
85 attachment.R = 1;
86 attachment.G = 1;
87 attachment.B = 1;
88 attachment.A = 1;
89
90 // pass OriginalWidth and OriginalHeight because UpdateOffset uses it in its calculation.
91 TextureRegion textreRegion = attachment.Region;
92 AtlasRegion atlasRegion = textreRegion as AtlasRegion;
93 float originalWidth = atlasRegion != null ? atlasRegion.originalWidth : textreRegion.width;
94 float originalHeight = atlasRegion != null ? atlasRegion.originalHeight : textreRegion.height;
95 attachment.Width = originalWidth * scale;
96 attachment.Height = originalHeight * scale;
97
98 attachment.SetColor(Color.white);
99 attachment.UpdateRegion();
100 return attachment;
101 }
