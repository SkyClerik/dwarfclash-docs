90 public void Initialize (bool overwrite = false) {
91 if (skeletonGraphic == null)
92 return;
93
94 skeletonGraphic.Initialize(false);
95
96 if (string.IsNullOrEmpty(slotName))
97 return;
98
99 // Don't reinitialize if the setup did not change.
100 if (!overwrite &&
101 colliderTable.Count > 0 && slot != null && // Slot is set and colliders already populated.
102 skeletonGraphic.Skeleton == slot.Skeleton && // Skeleton object did not change.
103 slotName == slot.Data.Name // Slot object did not change.
104 )
105 return;
106
107 slot = null;
108 currentAttachment = null;
109 currentAttachmentName = null;
110 currentCollider = null;
111 colliderTable.Clear();
112 nameTable.Clear();
113
114 Skeleton skeleton = skeletonGraphic.Skeleton;
115 if (skeleton == null)
116 return;
117 slot = skeleton.FindSlot(slotName);
118 if (slot == null) {
119 if (BoundingBoxFollowerGraphic.DebugMessages)
120 Debug.LogWarning(string.Format("Slot '{0}' not found for BoundingBoxFollowerGraphic on '{1}'. (Previous colliders were disposed.)", slotName, this.gameObject.name));
121 return;
122 }
123 int slotIndex = slot.Data.Index;
124
125 int requiredCollidersCount = 0;
126 PolygonCollider2D[] colliders = GetComponents<PolygonCollider2D>();
127 if (this.gameObject.activeInHierarchy) {
128 float scale = skeletonGraphic.MeshScale;
129 foreach (Skin skin in skeleton.Data.Skins)
130 AddCollidersForSkin(skin, slotIndex, colliders, scale, ref requiredCollidersCount);
131
132 if (skeleton.Skin != null)
133 AddCollidersForSkin(skeleton.Skin, slotIndex, colliders, scale, ref requiredCollidersCount);
134 }
135 DisposeExcessCollidersAfter(requiredCollidersCount);
136 skinBoneEnabled = slot.Bone.Active;
137
138 if (BoundingBoxFollowerGraphic.DebugMessages) {
139 bool valid = colliderTable.Count != 0;
140 if (!valid) {
141 if (this.gameObject.activeInHierarchy)
142 Debug.LogWarning("Bounding Box Follower not valid! Slot [" + slotName + "] does not contain any Bounding Box Attachments!");
143 else
144 Debug.LogWarning("Bounding Box Follower tried to rebuild as a prefab.");
145 }
146 }
147 }

149 void AddCollidersForSkin (Skin skin, int slotIndex, PolygonCollider2D[] previousColliders, float scale, ref int collidersCount) {
150 if (skin == null) return;
151 List<Skin.SkinEntry> skinEntries = new List<Skin.SkinEntry>();
152 skin.GetAttachments(slotIndex, skinEntries);
153
154 foreach (Skin.SkinEntry entry in skinEntries) {
155 Attachment attachment = skin.GetAttachment(slotIndex, entry.Name);
156 BoundingBoxAttachment boundingBoxAttachment = attachment as BoundingBoxAttachment;
157
158 if (BoundingBoxFollowerGraphic.DebugMessages && attachment != null && boundingBoxAttachment == null)
159 Debug.Log("BoundingBoxFollowerGraphic tried to follow a slot that contains non-boundingbox attachments: " + slotName);
160
161 if (boundingBoxAttachment != null) {
162 if (!colliderTable.ContainsKey(boundingBoxAttachment)) {
163 PolygonCollider2D bbCollider = collidersCount < previousColliders.Length ?
164 previousColliders[collidersCount] : gameObject.AddComponent<PolygonCollider2D>();
165 ++collidersCount;
166 SkeletonUtility.SetColliderPointsLocal(bbCollider, slot, boundingBoxAttachment, scale);
167 bbCollider.isTrigger = isTrigger;
168 bbCollider.usedByEffector = usedByEffector;
169 bbCollider.usedByComposite = usedByComposite;
170 bbCollider.enabled = false;
171 bbCollider.hideFlags = HideFlags.NotEditable;
172 colliderTable.Add(boundingBoxAttachment, bbCollider);
173 nameTable.Add(boundingBoxAttachment, entry.Name);
174 }
175 }
176 }

197 void DisposeExcessCollidersAfter (int requiredCount) {
198 PolygonCollider2D[] colliders = GetComponents<PolygonCollider2D>();
199 if (colliders.Length == 0) return;
200
201 for (int i = requiredCount; i < colliders.Length; ++i) {
202 PolygonCollider2D collider = colliders[i];
203 if (collider != null) {
204#if UNITY_EDITOR
205 if (Application.isEditor && !Application.isPlaying)
206 DestroyImmediate(collider);
207 else
208#endif
209 Destroy(collider);
210 }
211 }
212 }

214 void LateUpdate () {
215 if (slot != null && (slot.Attachment != currentAttachment || skinBoneEnabled != slot.Bone.Active)) {
216 skinBoneEnabled = slot.Bone.Active;
217 MatchAttachment(slot.Attachment);
218 }

223 void MatchAttachment (Attachment attachment) {
224 BoundingBoxAttachment bbAttachment = attachment as BoundingBoxAttachment;
225
226 if (BoundingBoxFollowerGraphic.DebugMessages && attachment != null && bbAttachment == null)
227 Debug.LogWarning("BoundingBoxFollowerGraphic tried to match a non-boundingbox attachment. It will treat it as null.");
228
229 if (currentCollider != null)
230 currentCollider.enabled = false;
231
232 if (bbAttachment == null) {
233 currentCollider = null;
234 currentAttachment = null;
235 currentAttachmentName = null;
236 } else {
237 PolygonCollider2D foundCollider;
238 colliderTable.TryGetValue(bbAttachment, out foundCollider);
239 if (foundCollider != null) {
240 currentCollider = foundCollider;
241 currentCollider.enabled = true;
242 currentAttachment = bbAttachment;
243 currentAttachmentName = nameTable[bbAttachment];
244 } else {
245 currentCollider = null;
246 currentAttachment = bbAttachment;
247 currentAttachmentName = null;
248 if (BoundingBoxFollowerGraphic.DebugMessages) Debug.LogFormat("Collider for BoundingBoxAttachment named '{0}' was not initialized. It is possibly from a new skin. currentAttachmentName will be null. You may need to call BoundingBoxFollowerGraphic.Initialize(overwrite: true);", bbAttachment.Name);
249 }
250 }
