46 // Job для сортировки
47 [BurstCompile]
48 private struct SortJob : IJob
49 {
50 public NativeArray<float> distances;
51 public NativeArray<Matrix4x4> matrices;
52 public int count;
53
54 public void Execute()
55 {
56 for (int i = 0; i < count - 1; i++)
57 {
58 for (int j = i + 1; j < count; j++)
59 {
60 if (distances[i] < distances[j])
61 {
62 float tempDist = distances[i];
63 distances[i] = distances[j];
64 distances[j] = tempDist;
65 Matrix4x4 tempMat = matrices[i];
66 matrices[i] = matrices[j];
67 matrices[j] = tempMat;
68 }
69 }
70 }
71 }

120 void Update()
121 {
122 if (meshRenderers.Count == 0) return;
123 if (cachedData.Length != meshRenderers.Count) InitializeCache();
124
125 Vector3 cameraPos = mainCamera.transform.position;
126 Plane[] frustumPlanes = GeometryUtility.CalculateFrustumPlanes(mainCamera);
127
128 frameCounter++;
129 bool updateBake = frameCounter % bakeUpdateFrameInterval == 0;
130
131 // Очищаем динамические группы (теперь правильный тип)
132 var dynamicGroups = GetLayerDictFromPool(); // Слой -> Материал -> Меш -> Список матриц
133
134 // Обновляем кеш и группируем
135 for (int i = 0; i < meshRenderers.Count; i++)
136 {
137 var mr = meshRenderers[i];
138 if (mr == null || mr.sharedMaterial == null || mr.GetComponent<MeshFilter>() == null) continue;
139
140 Vector3 pos = mr.transform.position;
141 Quaternion rot = mr.transform.rotation;
142 Mesh mesh = mr.GetComponent<MeshFilter>().sharedMesh;
143 Material mat = mr.sharedMaterial;
144 int layer = mr.gameObject.layer;
145
146 bool needsUpdate = Vector3.Distance(pos, cachedData[i].lastPosition) > movementThreshold ||
147 Quaternion.Angle(rot, cachedData[i].lastRotation) > rotationThreshold ||
148 mesh != cachedData[i].lastMesh || mat != cachedData[i].lastMaterial || layer != cachedData[i].lastLayer;
149
150 Bounds bounds = new Bounds(pos, Vector3.one * 1f);
151 bool isVisible = GeometryUtility.TestPlanesAABB(frustumPlanes, bounds);
152 cachedData[i].isVisible = isVisible;
153
154 if (needsUpdate)
155 {
156 cachedData[i].lastPosition = pos;
157 cachedData[i].lastRotation = rot;
158 cachedData[i].matrix = mr.transform.localToWorldMatrix;
159 cachedData[i].distance = Vector3.Distance(pos, cameraPos);
160 cachedData[i].lastMesh = mesh;
161 cachedData[i].lastMaterial = mat;
162 cachedData[i].lastLayer = layer;
163 cachedData[i].isBaked = false; // Сбрасываем bake
164 }
165
166 // Bake: Если объект статичный и видимый, добавляем в bakedGroups
167 if (!needsUpdate && isVisible && updateBake)
168 {
169 if (!bakedGroups.ContainsKey(layer))
170 bakedGroups[layer] = new Dictionary<Material, Dictionary<Mesh, List<Matrix4x4>>>();
171 if (!bakedGroups[layer].ContainsKey(mat))
172 bakedGroups[layer][mat] = new Dictionary<Mesh, List<Matrix4x4>>();
173 if (!bakedGroups[layer][mat].ContainsKey(mesh))
174 bakedGroups[layer][mat][mesh] = new List<Matrix4x4>();
175 bakedGroups[layer][mat][mesh].Add(cachedData[i].matrix);
176 cachedData[i].isBaked = true;
177 }
178 else if (isVisible && cachedData[i].matrix != Matrix4x4.zero)
179 {
180 // Динамическая группа (теперь правильно: dynamicGroups[layer][mat][mesh])
181 if (!dynamicGroups.ContainsKey(layer))
182 dynamicGroups[layer] = GetMatDictFromPool();
183 if (!dynamicGroups[layer].ContainsKey(mat))
184 dynamicGroups[layer][mat] = GetMeshDictFromPool();
185 if (!dynamicGroups[layer][mat].ContainsKey(mesh))
186 dynamicGroups[layer][mat][mesh] = GetListFromPool();
187 dynamicGroups[layer][mat][mesh].Add(cachedData[i].matrix);
188 }
189 }
190
191 // Рендерим bake (статичные)
192 RenderGroups(bakedGroups, false);
193
194 // Рендерим динамические
195 RenderGroups(dynamicGroups, true);
196
197 // Возвращаем в пул
198 ReturnLayerDictToPool(dynamicGroups);
199 }

201 private void RenderGroups(Dictionary<int, Dictionary<Material, Dictionary<Mesh, List<Matrix4x4>>>> groups, bool isDynamic)
202 {
203 foreach (var layerGroup in groups)
204 {
205 foreach (var matGroup in layerGroup.Value)
206 {
207 foreach (var meshGroup in matGroup.Value)
208 {
209 var list = meshGroup.Value;
210 if (list.Count == 0) continue;
211
212 // Сортировка по расстоянию (только для динамичных)
213 if (isDynamic)
214 {
215 NativeArray<float> distances = new NativeArray<float>(list.Count, Allocator.TempJob);
216 NativeArray<Matrix4x4> matrices = new NativeArray<Matrix4x4>(list.Count, Allocator.TempJob);
217 for (int i = 0; i < list.Count; i++)
218 {
219 matrices[i] = list[i];
220 distances[i] = 0; // Упрощённо; в реальности сопоставьте с cachedData
221 }
222 SortJob sortJob = new SortJob { distances = distances, matrices = matrices, count = list.Count };
223 JobHandle handle = sortJob.Schedule();
224 handle.Complete();
225 list.Clear();
226 for (int i = 0; i < matrices.Length; i++) list.Add(matrices[i]);
227 distances.Dispose();
228 matrices.Dispose();
229 }
230
231 // Рендерим батчами
232 int batchSize = 1023;
233 for (int i = 0; i < list.Count; i += batchSize)
234 {
235 int count = Mathf.Min(batchSize, list.Count - i);
236 Matrix4x4[] batchMatrices = new Matrix4x4[count];
237 for (int j = 0; j < count; j++) batchMatrices[j] = list[i + j];
238 Graphics.DrawMeshInstanced(meshGroup.Key, 0, matGroup.Key, batchMatrices);
239 }
240 }
241 }
242 }
