73 public static void UpdateAudioRoom(FmodResonanceAudioRoom room, bool roomEnabled)
74 {
75 // Update the enabled rooms list.
76 if (roomEnabled)
77 {
78 if (!enabledRooms.Contains(room))
79 {
80 enabledRooms.Add(room);
81 }
82 }
83 else
84 {
85 enabledRooms.Remove(room);
86 }
87 // Update the current room effects to be applied.
88 if (enabledRooms.Count > 0)
89 {
90 FmodResonanceAudioRoom currentRoom = enabledRooms[enabledRooms.Count - 1];
91 RoomProperties roomProperties = GetRoomProperties(currentRoom);
92 // Pass the room properties into a pointer.
93 IntPtr roomPropertiesPtr = Marshal.AllocHGlobal(roomPropertiesSize);
94 Marshal.StructureToPtr(roomProperties, roomPropertiesPtr, false);
95 ListenerPlugin.setParameterData(roomPropertiesIndex, GetBytes(roomPropertiesPtr,
96 roomPropertiesSize));
97 Marshal.FreeHGlobal(roomPropertiesPtr);
98 }
99 else
100 {
101 // Set the room properties to a null room, which will effectively disable the room effects.
102 ListenerPlugin.setParameterData(roomPropertiesIndex, GetBytes(IntPtr.Zero, 0));
103 }
104 }

107 public static bool IsListenerInsideRoom(FmodResonanceAudioRoom room)
108 {
109 // Compute the room position relative to the listener.
110 FMOD.VECTOR unused;
111 RuntimeManager.CoreSystem.get3DListenerAttributes(0, out listenerPositionFmod, out unused,
112 out unused, out unused);
113 Vector3 listenerPosition = new Vector3(listenerPositionFmod.x, listenerPositionFmod.y,
114 listenerPositionFmod.z);
115 Vector3 relativePosition = listenerPosition - room.transform.position;
116 Quaternion rotationInverse = Quaternion.Inverse(room.transform.rotation);
117 // Set the size of the room as the boundary and return whether the listener is inside.
118 bounds.size = Vector3.Scale(room.transform.lossyScale, room.Size);
119 return bounds.Contains(rotationInverse * relativePosition);
120 }

184 // Converts given |position| and |rotation| from Unity space to audio space.
185 private static void ConvertAudioTransformFromUnity(ref Vector3 position,
186 ref Quaternion rotation)
187 {
188 // Compose the transformation matrix.
189 Matrix4x4 transformMatrix = Matrix4x4.TRS(position, rotation, Vector3.one);
190 // Convert the transformation matrix from left-handed to right-handed.
191 transformMatrix = flipZ * transformMatrix * flipZ;
192 // Update |position| and |rotation| respectively.
193 position = transformMatrix.GetColumn(3);
194 rotation = Quaternion.LookRotation(transformMatrix.GetColumn(2), transformMatrix.GetColumn(1));
195 }

197 // Returns a byte array of |length| created from |ptr|.
198 private static byte[] GetBytes(IntPtr ptr, int length)
199 {
200 if (ptr != IntPtr.Zero)
201 {
202 byte[] byteArray = new byte[length];
203 Marshal.Copy(ptr, byteArray, 0, length);
204 return byteArray;
205 }
206 // Return an empty array if the pointer is null.
207 return new byte[1];
208 }

241 // Initializes and returns the FMOD Resonance Audio Listener Plugin.
242 private static FMOD.DSP Initialize()
243 {
244 // Search through all busses on in banks.
245 int numBanks = 0;
246 FMOD.DSP dsp = new FMOD.DSP();
247 FMOD.Studio.Bank[] banks = null;
248 RuntimeManager.StudioSystem.getBankCount(out numBanks);
249 RuntimeManager.StudioSystem.getBankList(out banks);
250 for (int currentBank = 0; currentBank < numBanks; ++currentBank)
251 {
252 int numBusses = 0;
253 FMOD.Studio.Bus[] busses = null;
254 banks[currentBank].getBusCount(out numBusses);
255 banks[currentBank].getBusList(out busses);
256 for (int currentBus = 0; currentBus < numBusses; ++currentBus)
257 {
258 // Make sure the channel group of the current bus is assigned properly.
259 string busPath = null;
260 busses[currentBus].getPath(out busPath);
261 RuntimeManager.StudioSystem.getBus(busPath, out busses[currentBus]);
262 busses[currentBus].lockChannelGroup();
263 RuntimeManager.StudioSystem.flushCommands();
264 FMOD.ChannelGroup channelGroup;
265 busses[currentBus].getChannelGroup(out channelGroup);
266 if (channelGroup.hasHandle())
267 {
268 int numDsps = 0;
269 channelGroup.getNumDSPs(out numDsps);
270 for (int currentDsp = 0; currentDsp < numDsps; ++currentDsp)
271 {
272 channelGroup.getDSP(currentDsp, out dsp);
273 string dspNameSb;
274 int unusedInt = 0;
275 uint unusedUint = 0;
276 dsp.getInfo(out dspNameSb, out unusedUint, out unusedInt, out unusedInt, out unusedInt);
277 if (dspNameSb.ToString().Equals(listenerPluginName) && dsp.hasHandle())
278 {
279 return dsp;
280 }
281 }
282 }
283 busses[currentBus].unlockChannelGroup();
284 }
285 }
286 RuntimeUtils.DebugLogError(listenerPluginName + " not found in the FMOD project.");
287 return dsp;
288 }
