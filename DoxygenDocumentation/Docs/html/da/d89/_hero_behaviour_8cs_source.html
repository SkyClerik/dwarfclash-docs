285 [ContextMenu("Log All States")] // Правая кнопка в Inspector'е -> "Log All States"
286 public void LogAllStates()
287 {
288 if (_animator == null || _animator.runtimeAnimatorController == null)
289 {
290 Debug.LogError("Animator или его контроллер не присвоен!");
291 return;
292 }
293
294 AnimatorController controller = _animator.runtimeAnimatorController as AnimatorController;
295 if (controller == null)
296 {
297 Debug.LogError("Controller не найден!");
298 return;
299 }
300
301 List<string> states = new List<string>();
302 // Обход всех слоёв (layers)
303 for (int layerIndex = 0; layerIndex < controller.layers.Length; layerIndex++)
304 {
305 string layerName = controller.layers[layerIndex].name;
306 AnimatorStateMachine stateMachine = controller.layers[layerIndex].stateMachine;
307 TraverseStateMachine(stateMachine, states, layerName); // Передаём layerName вместо index
308 }
309
310 Debug.Log("Все найденные States (используйте эти имена в Play()):");
311 foreach (string statePath in states)
312 {
313 Debug.Log(statePath);
314 }
315 }
316
317 private void TraverseStateMachine(AnimatorStateMachine stateMachine, List<string> states, string currentPath = "")
318 {
319 // States в текущей машине
320 foreach (ChildAnimatorState childState in stateMachine.states)
321 {
322 string stateName = childState.state.name; // Это именно имя state
323 string stateFullPath = string.IsNullOrEmpty(currentPath) ? stateName : $"{currentPath}/{stateName}";
324 states.Add(stateFullPath);
325 }
326
327 // Рекурсия в субмашинах (Nested State Machines)
328 foreach (ChildAnimatorStateMachine subMachine in stateMachine.stateMachines)
329 {
330 string subPath = string.IsNullOrEmpty(currentPath) ? subMachine.stateMachine.name : $"{currentPath}/{subMachine.stateMachine.name}";
331 TraverseStateMachine(subMachine.stateMachine, states, subPath);
332 }
