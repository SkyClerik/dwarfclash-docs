46 public void SpawnPlayerHeroes(BattleTable playerTable, float centerOffset, float cellSpacing)
47 {
48 int numRows = playerTable.Table.Length;
49 int numCols = playerTable.Table.Length > 0 ? playerTable.Table[0].HeroBoxes.Length : 0;
50
51 for (int r = 0; r < numRows; r++)
52 {
53 for (int c = 0; c < numCols; c++)
54 {
55 HeroDefinition heroDef = playerTable.Table[r].HeroBoxes[c]?.baseHeroDefenition;
56 if (heroDef != null)
57 {
58 float x = -(centerOffset + (numCols - 1 - c) * cellSpacing);
59 float z = (numRows - 1 - r) * cellSpacing;
60 Vector3 position = new Vector3(x, 0, z);
61
62 InstantiateHero(heroDef, position);
63 }
64 }
65 }
66 }
