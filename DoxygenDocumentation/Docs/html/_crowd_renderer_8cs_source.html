67 public void UpdateParts()
68 {
69 if (_skeletons == null || _skeletons.Count == 0 || _crowdShader == null) return;
70
71 // --- Step 1: Clear and Gather All Parts ---
72 foreach (var list in _partsByTexture.Values) list.Clear();
73
74 // Сортируем персонажей по их Z-позиции (от дальнего к ближнему)
75 _skeletons.Sort((a, b) => b.transform.position.z.CompareTo(a.transform.position.z));
76
77 foreach (var skeletonAnimation in _skeletons)
78 {
79 if (skeletonAnimation == null || !skeletonAnimation.gameObject.activeInHierarchy) continue;
80
81 var skeleton = skeletonAnimation.Skeleton;
82 skeleton.ScaleX = this.flipX ? -1f : 1f;
83 Debug.Log($"Processing skeleton: {skeletonAnimation.name}, DrawOrder.Count: {skeleton.DrawOrder.Count}");
84 for (int i = 0; i < skeleton.DrawOrder.Count; i++)
85 {
86 var slot = skeleton.DrawOrder.Items[i];
87 Spine.RegionAttachment regionAttachment = slot.Attachment as Spine.RegionAttachment;
88 Spine.MeshAttachment meshAttachment = slot.Attachment as Spine.MeshAttachment;
89
90 float[] uvs = null; // Объявляем здесь
91
92 if (regionAttachment == null && meshAttachment == null)
93 {
94 Debug.Log($"Slot {i}: Attachment is neither RegionAttachment nor MeshAttachment. Skipping.");
95 continue; // Skip non-renderable Attachments
96 }
97
98 float[] worldVertices = new float[8]; // Для RegionAttachment
99 float[] meshWorldVertices = null; // Для MeshAttachment
100 int meshVerticesLength = 0;
101
102 if (regionAttachment != null)
103 {
104 regionAttachment.ComputeWorldVertices(slot, worldVertices, 0);
105 uvs = regionAttachment.UVs; // Присваиваем uvs
106 }
107 else if (meshAttachment != null)
108 {
109 meshVerticesLength = meshAttachment.WorldVerticesLength;
110 meshWorldVertices = new float[meshVerticesLength];
111 meshAttachment.ComputeWorldVertices(slot, 0, meshVerticesLength, meshWorldVertices, 0);
112 uvs = meshAttachment.UVs; // Присваиваем uvs
113 }
114
115 float blx, bly, ulx, uly, urx, ury, brx, bry;
116 if (regionAttachment != null)
117 {
118 blx = worldVertices[0]; bly = worldVertices[1];
119 ulx = worldVertices[2]; uly = worldVertices[3];
120 urx = worldVertices[4]; ury = worldVertices[5];
121 brx = worldVertices[6]; bry = worldVertices[7];
122 }
123 else // meshAttachment != null
124 {
125 // Для MeshAttachment берем bounding box или усредняем
126 // Для простоты пока возьмем первые 4 вершины, если они есть
127 if (meshVerticesLength >= 8)
128 {
129 blx = meshWorldVertices[0]; bly = meshWorldVertices[1];
130 ulx = meshWorldVertices[2]; uly = meshWorldVertices[3];
131 urx = meshWorldVertices[4]; ury = meshWorldVertices[5];
132 brx = meshWorldVertices[6]; bry = meshWorldVertices[7];
133 }
134 else
135 {
136 // Если вершин меньше 4, то это не квад, и нужно будет пересмотреть логику
137 // Пока пропустим, чтобы не было ошибок
138 Debug.LogWarning($"Slot {i}: MeshAttachment has less than 4 vertices. Skipping for now.");
139 continue;
140 }
141 }
142
143 var skeletonTransform = skeletonAnimation.transform;
144
145 Vector3 localPos = new Vector3((blx + urx) / 2f, (bly + ury) / 2f, 0);
146 Vector2 side_right = new Vector2(brx - blx, bry - bly);
147 Vector2 side_up = new Vector2(ulx - blx, uly - bly);
148 float localRotation = Mathf.Atan2(side_right.y, side_right.x);
149 Vector2 localScale = new Vector2(side_right.magnitude, side_up.magnitude);
150
151 // Apply skeleton's internal flip
152
153 if (skeleton.ScaleX < 0)
154
155 {
156
157 localScale.x = -localScale.x;
158
159 // The localRotation is ~180 degrees off. Subtract PI to correct it.
160
161 localRotation -= Mathf.PI;
162
163 }
164 if (skeleton.ScaleY < 0)
165 {
166 localScale.y = -localScale.y;
167 }
168
169 Vector3 finalPosition = skeletonTransform.TransformPoint(localPos);
170 finalPosition.z = skeletonAnimation.transform.position.z + (localPos.z * skeletonTransform.lossyScale.z) + (i * -0.0001f);
171 float finalRotation = localRotation + skeletonTransform.rotation.eulerAngles.z * Mathf.Deg2Rad;
172 Vector2 finalScale = new Vector2(localScale.x * skeletonTransform.lossyScale.x, localScale.y * skeletonTransform.lossyScale.y);
173
174 Spine.AtlasRegion region = null;
175 if (regionAttachment != null) region = regionAttachment.Region as Spine.AtlasRegion;
176 else if (meshAttachment != null) region = meshAttachment.Region as Spine.AtlasRegion;
177
178 if (region == null)
179 {
180 Debug.Log($"Slot {i}: Region is null. Skipping.");
181 continue;
182 }
183
184 Texture texture = null;
185 Material material = region.page.rendererObject as Material;
186 if (material != null)
187 {
188 texture = material.mainTexture;
189 }
190 else
191 {
192 Texture tex = region.page.rendererObject as Texture;
193 if (tex != null)
194 {
195 texture = tex;
196 }
197 }
198 if (texture == null)
199 {
200 Debug.Log($"Slot {i}: Texture is null. Skipping.");
201 continue;
202 }
203
204 Vector4 uvRect;
205 if (regionAttachment != null)
206 {
207 uvs = regionAttachment.UVs; // Присваиваем uvs
208 if (region.rotate)
209 {
210 finalRotation -= 90f * Mathf.Deg2Rad;
211 finalScale = new Vector2(finalScale.y, finalScale.x);
212 uvRect = new Vector4(uvs[4], uvs[5], Mathf.Abs(uvs[0] - uvs[4]), Mathf.Abs(uvs[1] - uvs[5]));
213 }
214 else
215 {
216 uvRect = new Vector4(uvs[0], uvs[1], uvs[4] - uvs[0], uvs[3] - uvs[1]);
217 }
218 }
219 else // meshAttachment != null
220 {
221 // Для MeshAttachment UVs могут быть сложнее, но пока возьмем первые 4
222 uvs = meshAttachment.UVs; // Присваиваем uvs
223 if (region.rotate)
224 {
225 finalRotation -= 90f * Mathf.Deg2Rad;
226 finalScale = new Vector2(finalScale.y, finalScale.x);
227 uvRect = new Vector4(uvs[4], uvs[5], Mathf.Abs(uvs[0] - uvs[4]), Mathf.Abs(uvs[1] - uvs[5]));
228 }
229 else
230 {
231 uvRect = new Vector4(uvs[0], uvs[1], uvs[4] - uvs[0], uvs[3] - uvs[1]);
232 }
233 }
234 if (!_partsByTexture.ContainsKey(texture))
235 {
236 _partsByTexture[texture] = new List<BodyPartData>();
237 }
238
239
240 _partsByTexture[texture].Add(new BodyPartData
241 {
242 position = finalPosition,
243 rotation = finalRotation,
244 scale = finalScale,
245 uv0 = new Vector2(uvs[0], uvs[1]),
246 uv1 = new Vector2(uvs[2], uvs[3]),
247 uv2 = new Vector2(uvs[4], uvs[5]),
248 uv3 = new Vector2(uvs[6], uvs[7])
249 });
250 finalPosition.z += i * -0.0001f;
251 Debug.Log($"BodyPartData: Pos={finalPosition}, Rot={finalRotation}, Scale={finalScale}, UV={uvRect}");
252 }
253 }

256 public void CalculateAverageZ()
257 {
258 if (_skeletons == null || _skeletons.Count == 0)
259 {
260 AverageZ = 0;
261 return;
262 }
263
264 float totalZ = 0;
265 foreach (var skeletonAnimation in _skeletons)
266 {
267 if (skeletonAnimation != null)
268 {
269 totalZ += skeletonAnimation.transform.position.z;
270 }
271 }
272 AverageZ = totalZ / _skeletons.Count;

275 void LateUpdate()
276 {
277 // Логика отрисовки, которая осталась после перемещения сбора частей в UpdateParts()
278 // --- Step 2: Draw All Parts, Batched by Texture ---
279 _bufferIndex = 0; // Reset buffer index for this frame's drawing phase
280 foreach (var pair in _partsByTexture)
281 {
282 var texture = pair.Key;
283 var partsList = pair.Value;
284 if (partsList.Count == 0) continue;
285
286 // Get the next available buffer
287 var currentBuffer = _partsBuffers[_bufferIndex];
288 currentBuffer.SetData(partsList, 0, 0, partsList.Count);
289
290 // Get material
291 if (!_runtimeMaterials.ContainsKey(texture))
292 {
293 var newMat = new Material(_crowdShader);
294 newMat.SetTexture("_MainTex", texture);
295 _runtimeMaterials[texture] = newMat;
296 debugMaterial = newMat; // Присваиваем для отладки
297 }
298 var material = _runtimeMaterials[texture];
299
300 // Set buffer and draw
301 _mpb.SetBuffer("_Parts", currentBuffer);
302 Graphics.DrawMeshInstancedProcedural(
303 _quadMesh, 0, material,
304 new Bounds(Vector3.zero, new Vector3(1000f, 1000f, 1000f)),
305 partsList.Count, _mpb);
306
307 // Move to the next buffer for the next draw call
308 _bufferIndex = (_bufferIndex + 1) % 3;
309 }
310 }
