83 protected void SafeSubscribe (Spine.TrackEntry trackEntry, AnimationEventTypes eventsToWaitFor) {
84 if (trackEntry == null) {
85 // Break immediately if trackEntry is null.
86 Debug.LogWarning("TrackEntry was null. Coroutine will continue immediately.");
87 m_WasFired = true;
88 } else {
89 if ((eventsToWaitFor & AnimationEventTypes.Start) != 0)
90 trackEntry.Start += HandleComplete;
91 if ((eventsToWaitFor & AnimationEventTypes.Interrupt) != 0)
92 trackEntry.Interrupt += HandleComplete;
93 if ((eventsToWaitFor & AnimationEventTypes.End) != 0)
94 trackEntry.End += HandleComplete;
95 if ((eventsToWaitFor & AnimationEventTypes.Dispose) != 0)
96 trackEntry.Dispose += HandleComplete;
97 if ((eventsToWaitFor & AnimationEventTypes.Complete) != 0)
98 trackEntry.Complete += HandleComplete;
99 }
