182 public static void GetWorldToLocalMatrix (this Bone bone, out float ia, out float ib, out float ic, out float id) {
183 float a = bone.A, b = bone.B, c = bone.C, d = bone.D;
184 float invDet = 1 / (a * d - b * c);
185 ia = invDet * d;
186 ib = invDet * -b;
187 ic = invDet * -c;
188 id = invDet * a;
189 }

234 public static Vector2[] GetLocalVertices (this VertexAttachment va, Slot slot, Vector2[] buffer) {
235 int floatsCount = va.WorldVerticesLength;
236 int bufferTargetSize = floatsCount >> 1;
237 buffer = buffer ?? new Vector2[bufferTargetSize];
238 if (buffer.Length < bufferTargetSize) throw new System.ArgumentException(string.Format("Vector2 buffer too small. {0} requires an array of size {1}. Use the attachment's .WorldVerticesLength to get the correct size.", va.Name, floatsCount), "buffer");
239
240 if (va.Bones == null && slot.Deform.Count == 0) {
241 float[] localVerts = va.Vertices;
242 for (int i = 0; i < bufferTargetSize; i++) {
243 int j = i * 2;
244 buffer[i] = new Vector2(localVerts[j], localVerts[j + 1]);
245 }
246 } else {
247 float[] floats = new float[floatsCount];
248 va.ComputeWorldVertices(slot, floats);
249
250 Bone sb = slot.Bone;
251 float ia, ib, ic, id, bwx = sb.WorldX, bwy = sb.WorldY;
252 sb.GetWorldToLocalMatrix(out ia, out ib, out ic, out id);
253
254 for (int i = 0; i < bufferTargetSize; i++) {
255 int j = i * 2;
256 float x = floats[j] - bwx, y = floats[j + 1] - bwy;
257 buffer[i] = new Vector2(x * ia + y * ib, x * ic + y * id);
258 }
259 }
260
261 return buffer;
262 }

268 public static Vector2[] GetWorldVertices (this VertexAttachment a, Slot slot, Vector2[] buffer) {
269 int worldVertsLength = a.WorldVerticesLength;
270 int bufferTargetSize = worldVertsLength >> 1;
271 buffer = buffer ?? new Vector2[bufferTargetSize];
272 if (buffer.Length < bufferTargetSize) throw new System.ArgumentException(string.Format("Vector2 buffer too small. {0} requires an array of size {1}. Use the attachment's .WorldVerticesLength to get the correct size.", a.Name, worldVertsLength), "buffer");
273
274 float[] floats = new float[worldVertsLength];
275 a.ComputeWorldVertices(slot, floats);
276
277 for (int i = 0, n = worldVertsLength >> 1; i < n; i++) {
278 int j = i * 2;
279 buffer[i] = new Vector2(floats[j], floats[j + 1]);
280 }
281
282 return buffer;
283 }

308 public float a, b, c, d, x, y;

311 public static BoneMatrix CalculateSetupWorld (BoneData boneData) {
312 if (boneData == null)
313 return default(BoneMatrix);
314
315 // End condition: isRootBone
316 if (boneData.Parent == null)
317 return GetInheritedInternal(boneData, default(BoneMatrix));
318
319 BoneMatrix result = CalculateSetupWorld(boneData.Parent);
320 return GetInheritedInternal(boneData, result);
321 }
322
323 static BoneMatrix GetInheritedInternal (BoneData boneData, BoneMatrix parentMatrix) {
324 BoneData parent = boneData.Parent;
325 if (parent == null) return new BoneMatrix(boneData); // isRootBone
326
327 float pa = parentMatrix.a, pb = parentMatrix.b, pc = parentMatrix.c, pd = parentMatrix.d;
328 BoneMatrix result = default(BoneMatrix);
329 result.x = pa * boneData.X + pb * boneData.Y + parentMatrix.x;
330 result.y = pc * boneData.X + pd * boneData.Y + parentMatrix.y;
331
332 switch (boneData.Inherit) {
333 case Inherit.Normal: {
334 float rotationY = boneData.Rotation + 90 + boneData.ShearY;
335 float la = MathUtils.CosDeg(boneData.Rotation + boneData.ShearX) * boneData.ScaleX;
336 float lb = MathUtils.CosDeg(rotationY) * boneData.ScaleY;
337 float lc = MathUtils.SinDeg(boneData.Rotation + boneData.ShearX) * boneData.ScaleX;
338 float ld = MathUtils.SinDeg(rotationY) * boneData.ScaleY;
339 result.a = pa * la + pb * lc;
340 result.b = pa * lb + pb * ld;
341 result.c = pc * la + pd * lc;
342 result.d = pc * lb + pd * ld;
343 break;
344 }
345 case Inherit.OnlyTranslation: {
346 float rotationY = boneData.Rotation + 90 + boneData.ShearY;
347 result.a = MathUtils.CosDeg(boneData.Rotation + boneData.ShearX) * boneData.ScaleX;
348 result.b = MathUtils.CosDeg(rotationY) * boneData.ScaleY;
349 result.c = MathUtils.SinDeg(boneData.Rotation + boneData.ShearX) * boneData.ScaleX;
350 result.d = MathUtils.SinDeg(rotationY) * boneData.ScaleY;
351 break;
352 }
353 case Inherit.NoRotationOrReflection: {
354 float s = pa * pa + pc * pc, prx;
355 if (s > 0.0001f) {
356 s = Math.Abs(pa * pd - pb * pc) / s;
357 pb = pc * s;
358 pd = pa * s;
359 prx = MathUtils.Atan2(pc, pa) * MathUtils.RadDeg;
360 } else {
361 pa = 0;
362 pc = 0;
363 prx = 90 - MathUtils.Atan2(pd, pb) * MathUtils.RadDeg;
364 }
365 float rx = boneData.Rotation + boneData.ShearX - prx;
366 float ry = boneData.Rotation + boneData.ShearY - prx + 90;
367 float la = MathUtils.CosDeg(rx) * boneData.ScaleX;
368 float lb = MathUtils.CosDeg(ry) * boneData.ScaleY;
369 float lc = MathUtils.SinDeg(rx) * boneData.ScaleX;
370 float ld = MathUtils.SinDeg(ry) * boneData.ScaleY;
371 result.a = pa * la - pb * lc;
372 result.b = pa * lb - pb * ld;
373 result.c = pc * la + pd * lc;
374 result.d = pc * lb + pd * ld;
375 break;
376 }
377 case Inherit.NoScale:
378 case Inherit.NoScaleOrReflection: {
379 float cos = MathUtils.CosDeg(boneData.Rotation), sin = MathUtils.SinDeg(boneData.Rotation);
380 float za = pa * cos + pb * sin;
381 float zc = pc * cos + pd * sin;
382 float s = (float)Math.Sqrt(za * za + zc * zc);
383 if (s > 0.00001f)
384 s = 1 / s;
385 za *= s;
386 zc *= s;
387 s = (float)Math.Sqrt(za * za + zc * zc);
388 float r = MathUtils.PI / 2 + MathUtils.Atan2(zc, za);
389 float zb = MathUtils.Cos(r) * s;
390 float zd = MathUtils.Sin(r) * s;
391 float la = MathUtils.CosDeg(boneData.ShearX) * boneData.ScaleX;
392 float lb = MathUtils.CosDeg(90 + boneData.ShearY) * boneData.ScaleY;
393 float lc = MathUtils.SinDeg(boneData.ShearX) * boneData.ScaleX;
394 float ld = MathUtils.SinDeg(90 + boneData.ShearY) * boneData.ScaleY;
395 if (boneData.Inherit != Inherit.NoScaleOrReflection ? pa * pd - pb * pc < 0 : false) {
396 zb = -zb;
397 zd = -zd;
398 }
399 result.a = za * la + zb * lc;
400 result.b = za * lb + zb * ld;
401 result.c = zc * la + zd * lc;
402 result.d = zc * lb + zd * ld;
403 break;
404 }
405 }
406
407 return result;
408 }

411 public BoneMatrix (BoneData boneData) {
412 float rotationY = boneData.Rotation + 90 + boneData.ShearY;
413 float rotationX = boneData.Rotation + boneData.ShearX;
414
415 a = MathUtils.CosDeg(rotationX) * boneData.ScaleX;
416 c = MathUtils.SinDeg(rotationX) * boneData.ScaleX;
417 b = MathUtils.CosDeg(rotationY) * boneData.ScaleY;
418 d = MathUtils.SinDeg(rotationY) * boneData.ScaleY;
419 x = boneData.X;
420 y = boneData.Y;
421 }
422
424 public BoneMatrix (Bone bone) {
425 float rotationY = bone.Rotation + 90 + bone.ShearY;
426 float rotationX = bone.Rotation + bone.ShearX;
427
428 a = MathUtils.CosDeg(rotationX) * bone.ScaleX;
429 c = MathUtils.SinDeg(rotationX) * bone.ScaleX;
430 b = MathUtils.CosDeg(rotationY) * bone.ScaleY;
431 d = MathUtils.SinDeg(rotationY) * bone.ScaleY;
432 x = bone.X;
433 y = bone.Y;
434 }

436 public BoneMatrix TransformMatrix (BoneMatrix local) {
437 return new BoneMatrix {
438 a = this.a * local.a + this.b * local.c,
439 b = this.a * local.b + this.b * local.d,
440 c = this.c * local.a + this.d * local.c,
441 d = this.c * local.b + this.d * local.d,
442 x = this.a * local.x + this.b * local.y + this.x,
443 y = this.c * local.x + this.d * local.y + this.y
444 };
445 }
