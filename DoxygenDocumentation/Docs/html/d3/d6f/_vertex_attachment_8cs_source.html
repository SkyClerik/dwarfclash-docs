102 public virtual void ComputeWorldVertices (Slot slot, int start, int count, float[] worldVertices, int offset, int stride = 2) {
103 count = offset + (count >> 1) * stride;
104 ExposedList<float> deformArray = slot.deform;
105 float[] vertices = this.vertices;
106 int[] bones = this.bones;
107 if (bones == null) {
108 if (deformArray.Count > 0) vertices = deformArray.Items;
109 Bone bone = slot.bone;
110 float x = bone.worldX, y = bone.worldY;
111 float a = bone.a, b = bone.b, c = bone.c, d = bone.d;
112 for (int vv = start, w = offset; w < count; vv += 2, w += stride) {
113 float vx = vertices[vv], vy = vertices[vv + 1];
114 worldVertices[w] = vx * a + vy * b + x;
115 worldVertices[w + 1] = vx * c + vy * d + y;
116 }
117 return;
118 }
119 int v = 0, skip = 0;
120 for (int i = 0; i < start; i += 2) {
121 int n = bones[v];
122 v += n + 1;
123 skip += n;
124 }
125 Bone[] skeletonBones = slot.bone.skeleton.bones.Items;
126 if (deformArray.Count == 0) {
127 for (int w = offset, b = skip * 3; w < count; w += stride) {
128 float wx = 0, wy = 0;
129 int n = bones[v++];
130 n += v;
131 for (; v < n; v++, b += 3) {
132 Bone bone = skeletonBones[bones[v]];
133 float vx = vertices[b], vy = vertices[b + 1], weight = vertices[b + 2];
134 wx += (vx * bone.a + vy * bone.b + bone.worldX) * weight;
135 wy += (vx * bone.c + vy * bone.d + bone.worldY) * weight;
136 }
137 worldVertices[w] = wx;
138 worldVertices[w + 1] = wy;
139 }
140 } else {
141 float[] deform = deformArray.Items;
142 for (int w = offset, b = skip * 3, f = skip << 1; w < count; w += stride) {
143 float wx = 0, wy = 0;
144 int n = bones[v++];
145 n += v;
146 for (; v < n; v++, b += 3, f += 2) {
147 Bone bone = skeletonBones[bones[v]];
148 float vx = vertices[b] + deform[f], vy = vertices[b + 1] + deform[f + 1], weight = vertices[b + 2];
149 wx += (vx * bone.a + vy * bone.b + bone.worldX) * weight;
150 wy += (vx * bone.c + vy * bone.d + bone.worldY) * weight;
151 }
152 worldVertices[w] = wx;
153 worldVertices[w + 1] = wy;
154 }
155 }
156 }
157 }
