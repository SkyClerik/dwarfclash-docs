260 public static Spine.Attachment GetAttachment (string attachmentPath, Spine.SkeletonData skeletonData) {
261 SpineAttachment.Hierarchy hierarchy = SpineAttachment.GetHierarchy(attachmentPath);
262 if (string.IsNullOrEmpty(hierarchy.name)) return null;
263
264 SlotData slot = skeletonData.FindSlot(hierarchy.slot);
265 if (slot == null) return null;
266 return skeletonData.FindSkin(hierarchy.skin).GetAttachment(slot.Index, hierarchy.name);
267 }
268
269 public static Spine.Attachment GetAttachment (string attachmentPath, SkeletonDataAsset skeletonDataAsset) {
270 return GetAttachment(attachmentPath, skeletonDataAsset.GetSkeletonData(true));
271 }

275 public struct Hierarchy {
276 public string skin;
277 public string slot;
278 public string name;
279
280 public Hierarchy (string fullPath) {
281 string[] chunks = fullPath.Split(new char[] { '/' }, System.StringSplitOptions.RemoveEmptyEntries);
282 if (chunks.Length == 0) {
283 skin = "";
284 slot = "";
285 name = "";
286 return;
287 } else if (chunks.Length < 2) {
288 throw new System.Exception("Cannot generate Attachment Hierarchy from string! Not enough components! [" + fullPath + "]");
289 }
290 skin = chunks[0];
291 slot = chunks[1];
292 name = "";
293 for (int i = 2; i < chunks.Length; i++) {
294 name += chunks[i];
295 }
296 }
297 }
298 }
