156 public SkeletonData GetSkeletonData (bool quiet) {
157 if (skeletonJSON == null) {
158#if UNITY_EDITOR
159 if (!errorIfSkeletonFileNullGlobal) quiet = true;
160#endif
161 if (!quiet)
162 Debug.LogError("Skeleton JSON file not set for SkeletonData asset: " + name, this);
163 Clear();
164 return null;
165 }
166
167 // Disabled to support attachmentless/skinless SkeletonData.
168 // if (atlasAssets == null) {
169 // atlasAssets = new AtlasAsset[0];
170 // if (!quiet)
171 // Debug.LogError("Atlas not set for SkeletonData asset: " + name, this);
172 // Clear();
173 // return null;
174 // }
175 // #if !SPINE_TK2D
176 // if (atlasAssets.Length == 0) {
177 // Clear();
178 // return null;
179 // }
180 // #else
181 // if (atlasAssets.Length == 0 && spriteCollection == null) {
182 // Clear();
183 // return null;
184 // }
185 // #endif
186
187 if (skeletonData != null)
188 return skeletonData;
189
190 AttachmentLoader attachmentLoader;
191 float skeletonDataScale;
192 Atlas[] atlasArray = this.GetAtlasArray();
193
194#if !SPINE_TK2D
195 attachmentLoader = (atlasArray.Length == 0) ? (AttachmentLoader)new RegionlessAttachmentLoader() : (AttachmentLoader)new AtlasAttachmentLoader(atlasArray);
196 skeletonDataScale = scale;
197#else
198 if (spriteCollection != null) {
199 attachmentLoader = new Spine.Unity.TK2D.SpriteCollectionAttachmentLoader(spriteCollection);
200 skeletonDataScale = (1.0f / (spriteCollection.invOrthoSize * spriteCollection.halfTargetHeight) * scale);
201 } else {
202 if (atlasArray.Length == 0) {
203 Reset();
204 if (!quiet) Debug.LogError("Atlas not set for SkeletonData asset: " + name, this);
205 return null;
206 }
207 attachmentLoader = new AtlasAttachmentLoader(atlasArray);
208 skeletonDataScale = scale;
209 }
210#endif
211
212 bool hasBinaryExtension = skeletonJSON.name.ToLower().Contains(".skel");
213 SkeletonData loadedSkeletonData = null;
214
215 try {
216 if (hasBinaryExtension) {
217#if UNSAFE_DIRECT_ACCESS_TEXT_ASSET_DATA
218 using (Stream stream = skeletonJSON.GetStreamUnsafe()) {
219 loadedSkeletonData = SkeletonDataAsset.ReadSkeletonData(stream, attachmentLoader, skeletonDataScale);
220 }
221#else
222 loadedSkeletonData = SkeletonDataAsset.ReadSkeletonData(skeletonJSON.bytes, attachmentLoader, skeletonDataScale);
223#endif
224 } else
225 loadedSkeletonData = SkeletonDataAsset.ReadSkeletonData(skeletonJSON.text, attachmentLoader, skeletonDataScale);
226 } catch (Exception ex) {
227 if (!quiet)
228 Debug.LogError("Error reading skeleton JSON file for SkeletonData asset: " + name + "\n" + ex.Message + "\n" + ex.StackTrace, skeletonJSON);
229 }
230
231#if UNITY_EDITOR
232 if (loadedSkeletonData == null && !quiet && skeletonJSON != null) {
233 string problemDescription = null;
234 bool isSpineSkeletonData;
235 SkeletonDataCompatibility.VersionInfo fileVersion = SkeletonDataCompatibility.GetVersionInfo(skeletonJSON, out isSpineSkeletonData, ref problemDescription);
236 if (problemDescription != null) {
237 if (!quiet)
238 Debug.LogError(problemDescription, skeletonJSON);
239 return null;
240 }
241 CompatibilityProblemInfo compatibilityProblemInfo = SkeletonDataCompatibility.GetCompatibilityProblemInfo(fileVersion);
242 if (compatibilityProblemInfo != null) {
243 SkeletonDataCompatibility.DisplayCompatibilityProblem(compatibilityProblemInfo.DescriptionString(), skeletonJSON);
244 return null;
245 }
246 }
247#endif
248 if (loadedSkeletonData == null)
249 return null;
250
251 if (skeletonDataModifiers != null) {
252 foreach (SkeletonDataModifierAsset modifier in skeletonDataModifiers) {
253 if (modifier != null && !(isUpgradingBlendModeMaterials && modifier is BlendModeMaterialsAsset)) {
254 modifier.Apply(loadedSkeletonData);
255 }
256 }
257 }
258 if (!isUpgradingBlendModeMaterials)
259 blendModeMaterials.ApplyMaterials(loadedSkeletonData);
260
261 this.InitializeWithData(loadedSkeletonData);
262
263 return skeletonData;
264 }

272 public void FillStateData (bool quiet = false) {
273 if (stateData != null) {
274 stateData.DefaultMix = defaultMix;
275
276 for (int i = 0, n = fromAnimation.Length; i < n; i++) {
277 string fromAnimationName = fromAnimation[i];
278 string toAnimationName = toAnimation[i];
279 if (fromAnimationName.Length == 0 || toAnimationName.Length == 0)
280 continue;
281#if UNITY_EDITOR
282 if (skeletonData.FindAnimation(fromAnimationName) == null) {
283 if (!quiet) Debug.LogError(
284 string.Format("Custom Mix Durations: Animation '{0}' not found, was it renamed?",
285 fromAnimationName), this);
286 continue;
287 }
288 if (skeletonData.FindAnimation(toAnimationName) == null) {
289 if (!quiet) Debug.LogError(
290 string.Format("Custom Mix Durations: Animation '{0}' not found, was it renamed?",
291 toAnimationName), this);
292 continue;
293 }
294#endif
295 stateData.SetMix(fromAnimationName, toAnimationName, duration[i]);
296 }
297 }

300 internal Atlas[] GetAtlasArray () {
301 List<Atlas> returnList = new System.Collections.Generic.List<Atlas>(atlasAssets.Length);
302 for (int i = 0; i < atlasAssets.Length; i++) {
303 AtlasAssetBase aa = atlasAssets[i];
304 if (aa == null) continue;
305 Atlas a = aa.GetAtlas();
306 if (a == null) continue;
307 returnList.Add(a);
308 }
309 return returnList.ToArray();
310 }
