281 static float[] pathVertexBuffer;
282 public static void DrawPath (Slot s, PathAttachment p, Transform t, bool includeName) {
283 if (UnityEngine.Event.current.type != EventType.Repaint) return;
284
285 int worldVerticesLength = p.WorldVerticesLength;
286
287 if (pathVertexBuffer == null || pathVertexBuffer.Length < worldVerticesLength)
288 pathVertexBuffer = new float[worldVerticesLength];
289
290 float[] pv = pathVertexBuffer;
291 p.ComputeWorldVertices(s, pv);
292
293 Color ocolor = Handles.color;
294 Handles.color = SpineHandles.PathColor;
295
296 Matrix4x4 m = t.localToWorldMatrix;
297 const int step = 6;
298 int n = worldVerticesLength - step;
299 Vector3 p0, p1, p2, p3;
300 for (int i = 2; i < n; i += step) {
301 p0 = m.MultiplyPoint(new Vector3(pv[i], pv[i + 1]));
302 p1 = m.MultiplyPoint(new Vector3(pv[i + 2], pv[i + 3]));
303 p2 = m.MultiplyPoint(new Vector3(pv[i + 4], pv[i + 5]));
304 p3 = m.MultiplyPoint(new Vector3(pv[i + 6], pv[i + 7]));
305 DrawCubicBezier(p0, p1, p2, p3);
306 }
307
308 n += step;
309 if (p.Closed) {
310 p0 = m.MultiplyPoint(new Vector3(pv[n - 4], pv[n - 3]));
311 p1 = m.MultiplyPoint(new Vector3(pv[n - 2], pv[n - 1]));
312 p2 = m.MultiplyPoint(new Vector3(pv[0], pv[1]));
313 p3 = m.MultiplyPoint(new Vector3(pv[2], pv[3]));
314 DrawCubicBezier(p0, p1, p2, p3);
315 }
316
317 const float endCapSize = 0.05f;
318 Vector3 firstPoint = m.MultiplyPoint(new Vector3(pv[2], pv[3]));
319 SpineHandles.DrawDot(firstPoint, endCapSize);
320
321 //if (!p.Closed) SpineHandles.DrawDot(m.MultiplyPoint(new Vector3(pv[n - 4], pv[n - 3])), endCapSize);
322 if (includeName) Handles.Label(firstPoint + new Vector3(0, 0.1f), p.Name, PathNameStyle);
323
324 Handles.color = ocolor;
325 }

342 public static void DrawBoundingBox (Slot slot, BoundingBoxAttachment box, Transform t) {
343 if (UnityEngine.Event.current.type != EventType.Repaint) return;
344
345 if (box.Vertices.Length <= 2) return; // Handle cases where user creates a BoundingBoxAttachment but doesn't actually define it.
346
347 float[] worldVerts = new float[box.WorldVerticesLength];
348 box.ComputeWorldVertices(slot, worldVerts);
349
350 Handles.color = Color.green;
351 Vector3 lastVert = Vector3.zero;
352 Vector3 vert = Vector3.zero;
353 Vector3 firstVert = t.TransformPoint(new Vector3(worldVerts[0], worldVerts[1], 0));
354 for (int i = 0; i < worldVerts.Length; i += 2) {
355 vert.x = worldVerts[i];
356 vert.y = worldVerts[i + 1];
357 vert.z = 0;
358
359 vert = t.TransformPoint(vert);
360
361 if (i > 0)
362 Handles.DrawLine(lastVert, vert);
363
364 lastVert = vert;
365 }
366
367 Handles.DrawLine(lastVert, firstVert);
368 }

385 public static void DrawConstraints (Transform transform, Skeleton skeleton, float skeletonRenderScale = 1f,
386 Vector2? positionOffset = null) {
387 if (UnityEngine.Event.current.type != EventType.Repaint) return;
388
389 Vector2 offset = positionOffset == null ? Vector2.zero : positionOffset.Value;
390 Vector3 targetPos;
391 Vector3 pos;
392 bool active;
393 Color handleColor;
394 const float Thickness = 4f;
395 Vector3 normal = transform.forward;
396
397 // Transform Constraints
398 handleColor = SpineHandles.TransformContraintColor;
399 foreach (TransformConstraint tc in skeleton.TransformConstraints) {
400 Bone targetBone = tc.Target;
401 targetPos = targetBone.GetWorldPosition(transform, skeletonRenderScale, offset);
402
403 if (tc.MixX > 0 || tc.MixY > 0) {
404 if ((tc.MixX > 0 && tc.MixX != 1f) ||
405 (tc.MixY > 0 && tc.MixY != 1f)) {
406 Handles.color = handleColor;
407 foreach (Bone b in tc.Bones) {
408 pos = b.GetWorldPosition(transform, skeletonRenderScale, offset);
409 Handles.DrawDottedLine(targetPos, pos, Thickness);
410 }
411 }
412 SpineHandles.DrawBoneCircle(targetPos, handleColor, normal, 1.3f * skeletonRenderScale);
413 Handles.color = handleColor;
414 SpineHandles.DrawCrosshairs(targetPos, 0.2f, targetBone.A, targetBone.B, targetBone.C, targetBone.D, transform, skeletonRenderScale);
415 }
416 }
417
418 // IK Constraints
419 handleColor = SpineHandles.IkColor;
420 foreach (IkConstraint ikc in skeleton.IkConstraints) {
421 Bone targetBone = ikc.Target;
422 targetPos = targetBone.GetWorldPosition(transform, skeletonRenderScale, offset);
423 ExposedList<Bone> bones = ikc.Bones;
424 active = ikc.Mix > 0;
425 if (active) {
426 pos = bones.Items[0].GetWorldPosition(transform, skeletonRenderScale, offset);
427 switch (bones.Count) {
428 case 1: {
429 Handles.color = handleColor;
430 Handles.DrawLine(targetPos, pos);
431 SpineHandles.DrawBoneCircle(targetPos, handleColor, normal);
432 Matrix4x4 m = bones.Items[0].GetMatrix4x4();
433 m.m03 = targetBone.WorldX * skeletonRenderScale + offset.x;
434 m.m13 = targetBone.WorldY * skeletonRenderScale + offset.y;
435 SpineHandles.DrawArrowhead(transform.localToWorldMatrix * m);
436 break;
437 }
438 case 2: {
439 Bone childBone = bones.Items[1];
440 Vector3 child = childBone.GetWorldPosition(transform, skeletonRenderScale, offset);
441 Handles.color = handleColor;
442 Handles.DrawLine(child, pos);
443 Handles.DrawLine(targetPos, child);
444 SpineHandles.DrawBoneCircle(pos, handleColor, normal, 0.5f);
445 SpineHandles.DrawBoneCircle(child, handleColor, normal, 0.5f);
446 SpineHandles.DrawBoneCircle(targetPos, handleColor, normal);
447 Matrix4x4 m = childBone.GetMatrix4x4();
448 m.m03 = targetBone.WorldX * skeletonRenderScale + offset.x;
449 m.m13 = targetBone.WorldY * skeletonRenderScale + offset.y;
450 SpineHandles.DrawArrowhead(transform.localToWorldMatrix * m);
451 break;
452 }
453 }
454 }
455 //Handles.Label(targetPos, ikc.Data.Name, SpineHandles.BoneNameStyle);
456 }
457
458 // Path Constraints
459 handleColor = SpineHandles.PathColor;
460 foreach (PathConstraint pc in skeleton.PathConstraints) {
461 active = pc.MixX > 0 || pc.MixY > 0 || pc.MixRotate > 0;
462 if (active)
463 foreach (Bone b in pc.Bones)
464 SpineHandles.DrawBoneCircle(b.GetWorldPosition(transform, skeletonRenderScale, offset),
465 handleColor, normal, 1f * skeletonRenderScale);
466 }
467 }

579 static void DrawArrowhead (Matrix4x4 m) {
580 if (UnityEngine.Event.current.type != EventType.Repaint) return;
581
