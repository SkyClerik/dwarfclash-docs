59 void OnEnable () {
60 skeletonUtility = (SkeletonUtility)target;
61 skeletonRenderer = skeletonUtility.skeletonRenderer;
62 skeletonGraphic = skeletonUtility.skeletonGraphic;
63 skeleton = skeletonUtility.Skeleton;
64
65 if (skeleton == null) {
66 if (skeletonRenderer != null) {
67 skeletonRenderer.Initialize(false);
68 skeletonRenderer.LateUpdate();
69 } else if (skeletonGraphic != null) {
70 skeletonGraphic.Initialize(false);
71 skeletonGraphic.LateUpdate();
72 }
73 skeleton = skeletonUtility.Skeleton;
74 }
75
76 if ((skeletonRenderer != null && !skeletonRenderer.valid) ||
77 (skeletonGraphic != null && !skeletonGraphic.IsValid)) return;
78
79#if !NEW_PREFAB_SYSTEM
80 isPrefab |= PrefabUtility.GetPrefabType(this.target) == PrefabType.Prefab;
81#endif
82 }

84 public override void OnInspectorGUI () {
85
86#if !NEW_PREFAB_SYSTEM
87 if (isPrefab) {
88 GUILayout.Label(new GUIContent("Cannot edit Prefabs", Icons.warning));
89 return;
90 }

93 serializedObject.Update();
94
95 if ((skeletonRenderer != null && !skeletonRenderer.valid) ||
96 (skeletonGraphic != null && !skeletonGraphic.IsValid)) {
97 GUILayout.Label(new GUIContent("Spine Component invalid. Check SkeletonData asset.", Icons.warning));
98 return;

101 EditorGUILayout.PropertyField(serializedObject.FindProperty("boneRoot"), SpineInspectorUtility.TempContent("Skeleton Root"));
102 EditorGUILayout.PropertyField(serializedObject.FindProperty("flipBy180DegreeRotation"), SpineInspectorUtility.TempContent("Flip by Rotation", null,
103 "If true, Skeleton.ScaleX and Skeleton.ScaleY are followed " +
104 "by 180 degree rotation. If false, negative Transform scale is used. " +
105 "Note that using negative scale is consistent with previous behaviour (hence the default), " +
106 "however causes serious problems with rigidbodies and physics. Therefore, it is recommended to " +
107 "enable this parameter where possible. When creating hinge chains for a chain of skeleton bones " +
108 "via SkeletonUtilityBone, it is mandatory to have this parameter enabled."));
109
110 bool hasRootBone = skeletonUtility.boneRoot != null;
111
112 if (!hasRootBone)
113 EditorGUILayout.HelpBox("No hierarchy found. Use Spawn Hierarchy to generate GameObjects for bones.", MessageType.Info);
114
115 using (new EditorGUI.DisabledGroupScope(hasRootBone)) {
116 if (SpineInspectorUtility.LargeCenteredButton(SpawnHierarchyButtonLabel))
117 SpawnHierarchyContextMenu();
118 }
119
120 if (hasRootBone) {
121 if (SpineInspectorUtility.CenteredButton(new GUIContent("Remove Hierarchy"))) {
122 Undo.RegisterCompleteObjectUndo(skeletonUtility, "Remove Hierarchy");
123 Undo.DestroyObjectImmediate(skeletonUtility.boneRoot.gameObject);
124 skeletonUtility.boneRoot = null;
125 }
126 }

128 serializedObject.ApplyModifiedProperties();

143 public static void AttachIcon (SkeletonUtilityBone boneComponent) {
144 Skeleton skeleton = boneComponent.hierarchy.Skeleton;
145 Texture2D icon = boneComponent.bone.Data.Length == 0 ? Icons.nullBone : Icons.boneNib;
146
147 foreach (IkConstraint c in skeleton.IkConstraints)
148 if (c.Target == boneComponent.bone) {
149 icon = Icons.constraintNib;
150 break;
151 }
152#if PUBLIC_SET_ICON_FOR_OBJECT
153 EditorGUIUtility.SetIconForObject(boneComponent.gameObject, icon);
154#else
155 typeof(EditorGUIUtility).InvokeMember("SetIconForObject", BindingFlags.InvokeMethod | BindingFlags.Static | BindingFlags.NonPublic, null, null, new object[2] {
156 boneComponent.gameObject,
157 icon
158 });
159#endif
160 }

162 static void AttachIconsToChildren (Transform root) {
163 if (root != null) {
164 SkeletonUtilityBone[] utilityBones = root.GetComponentsInChildren<SkeletonUtilityBone>();
165 foreach (SkeletonUtilityBone utilBone in utilityBones)
166 AttachIcon(utilBone);
167 }
