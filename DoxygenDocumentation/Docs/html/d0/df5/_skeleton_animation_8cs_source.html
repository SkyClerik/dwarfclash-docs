118 public string AnimationName {
119 get {
120 if (!valid) {
121 return _animationName;
122 } else {
123 TrackEntry entry = state.GetCurrent(0);
124 return entry == null ? null : entry.Animation.Name;
125 }
126 }
127 set {
128 Initialize(false);
129 if (_animationName == value) {
130 TrackEntry entry = state.GetCurrent(0);
131 if (entry != null && entry.Loop == loop)
132 return;
133 }
134 _animationName = value;
135
136 if (string.IsNullOrEmpty(value)) {
137 state.ClearTrack(0);
138 } else {
139 Spine.Animation animationObject = skeletonDataAsset.GetSkeletonData(false).FindAnimation(value);
140 if (animationObject != null)
141 state.SetAnimation(0, animationObject, loop);
142 }
143 }

173 public override void ClearState () {

181 public override void Initialize (bool overwrite, bool quiet = false) {
182 if (valid && !overwrite)
183 return;
184#if UNITY_EDITOR
185 if (BuildUtilities.IsInSkeletonAssetBuildPreProcessing)
186 return;
187#endif
188 state = null; // prevent applying leftover AnimationState
189 base.Initialize(overwrite, quiet);
190
191 if (!valid)
192 return;
193 state = new Spine.AnimationState(skeletonDataAsset.GetAnimationStateData());
194 state.Dispose += OnAnimationDisposed;
195 wasUpdatedAfterInit = false;
196
197 if (!string.IsNullOrEmpty(_animationName)) {
198 Spine.Animation animationObject = skeletonDataAsset.GetSkeletonData(false).FindAnimation(_animationName);
199 if (animationObject != null) {
200 state.SetAnimation(0, animationObject, loop);
201#if UNITY_EDITOR
202 if (!Application.isPlaying)
203 Update(0f);
204#endif
205 }
206 }
207
208 if (_OnAnimationRebuild != null)
209 _OnAnimationRebuild(this);
210 }

212 virtual protected void Update () {
213#if UNITY_EDITOR
214 if (!Application.isPlaying) {
215 Update(0f);
216 return;
217 }
218#endif
219 if (updateTiming != UpdateTiming.InUpdate) return;
220 Update(unscaledTime ? Time.unscaledDeltaTime : Time.deltaTime);
221 }
222
223 virtual protected void FixedUpdate () {
224 if (updateTiming != UpdateTiming.InFixedUpdate) return;
225 Update(unscaledTime ? Time.unscaledDeltaTime : Time.deltaTime);
226 }

229 public void Update (float deltaTime) {
230 if (!valid || state == null)
231 return;
232
233 wasUpdatedAfterInit = true;
234 if (updateMode < UpdateMode.OnlyAnimationStatus)
235 return;
236 UpdateAnimationStatus(deltaTime);
237
238 if (updateMode == UpdateMode.OnlyAnimationStatus)
239 return;
240 ApplyAnimation();
241 }
242
243 protected void UpdateAnimationStatus (float deltaTime) {
244 deltaTime *= timeScale;
245 state.Update(deltaTime);
246 skeleton.Update(deltaTime);
247
248 ApplyTransformMovementToPhysics();
249
250 if (updateMode == UpdateMode.OnlyAnimationStatus) {
251 state.ApplyEventTimelinesOnly(skeleton, issueEvents: false);
252 return;
253 }
254 }
255
256 public virtual void ApplyAnimation () {
257 if (_BeforeApply != null)
258 _BeforeApply(this);
259
260 if (updateMode != UpdateMode.OnlyEventTimelines)
261 state.Apply(skeleton);
262 else
263 state.ApplyEventTimelinesOnly(skeleton, issueEvents: true);
264
265 AfterAnimationApplied();
266 }
267
268 public void AfterAnimationApplied () {
269 if (_UpdateLocal != null)
270 _UpdateLocal(this);
271
272 if (_UpdateWorld == null) {
273 UpdateWorldTransform(Skeleton.Physics.Update);
274 } else {
275 UpdateWorldTransform(Skeleton.Physics.Pose);
276 _UpdateWorld(this);
277 UpdateWorldTransform(Skeleton.Physics.Update);
278 }
279
280 if (_UpdateComplete != null) {
281 _UpdateComplete(this);
282 }
283 }
284
285 public override void LateUpdate () {
286 if (updateTiming == UpdateTiming.InLateUpdate && valid)
287 Update(unscaledTime ? Time.unscaledDeltaTime : Time.deltaTime);
288
289 // instantiation can happen from Update() after this component, leading to a missing Update() call.
290 if (!wasUpdatedAfterInit) Update(0);
291
292 base.LateUpdate();
293 }
294
295 public override void OnBecameVisible () {
296 UpdateMode previousUpdateMode = updateMode;
297 updateMode = UpdateMode.FullUpdate;
298
299 // OnBecameVisible is called after LateUpdate()
300 if (previousUpdateMode != UpdateMode.FullUpdate &&
301 previousUpdateMode != UpdateMode.EverythingExceptMesh)
302 Update(0);
303 if (previousUpdateMode != UpdateMode.FullUpdate)
304 LateUpdate();
305 }
306
307 protected virtual void OnAnimationDisposed (TrackEntry entry) {
308 // when updateMode disables applying animations, still ensure animations are mixed out
309 if (updateMode != UpdateMode.FullUpdate &&
310 updateMode != UpdateMode.EverythingExceptMesh) {
311 entry.Animation.Apply(skeleton, 0, 0, false, null, 0f, MixBlend.Setup, MixDirection.Out);
312 }
313 }
