132 public override void OnInspectorGUI()
133 {
134 serializedObject.Update();
135
136 // Add clickable script field, as would have been provided by DrawDefaultInspector()
137 MonoScript script = MonoScript.FromMonoBehaviour(target as MonoBehaviour);
138 EditorGUI.BeginDisabledGroup(true);
139 EditorGUILayout.ObjectField("Script", script, typeof(MonoScript), false);
140 EditorGUI.EndDisabledGroup();
141 EditorGUILayout.LabelField(surfaceMaterialsLabel);
142 ++EditorGUI.indentLevel;
143 DrawSurfaceMaterial(leftWall, L10n.Tr("Left Wall"));
144 DrawSurfaceMaterial(rightWall, L10n.Tr("Right Wall"));
145 DrawSurfaceMaterial(floor, L10n.Tr("Floor"));
146 DrawSurfaceMaterial(ceiling, L10n.Tr("Ceiling"));
147 DrawSurfaceMaterial(backWall, L10n.Tr("Back Wall"));
148 DrawSurfaceMaterial(frontWall, L10n.Tr("Front Wall"));
149 --EditorGUI.indentLevel;
150
151 EditorGUILayout.Separator();
152
153 EditorGUILayout.Slider(reflectivity, 0.0f, FmodResonanceAudio.MaxReflectivity, reflectivityLabel);
154
155 EditorGUILayout.Separator();
156
157 EditorGUILayout.LabelField(reverbPropertiesLabel);
158 ++EditorGUI.indentLevel;
159 EditorGUILayout.Slider(reverbGainDb, FmodResonanceAudio.MinGainDb, FmodResonanceAudio.MaxGainDb,
160 reverbGainLabel);
161 EditorGUILayout.Slider(reverbBrightness, FmodResonanceAudio.MinReverbBrightness,
162 FmodResonanceAudio.MaxReverbBrightness, reverbBrightnessLabel);
163 EditorGUILayout.Slider(reverbTime, 0.0f, FmodResonanceAudio.MaxReverbTime, reverbTimeLabel);
164 --EditorGUI.indentLevel;
165
166 EditorGUILayout.Separator();
167
168 EditorGUILayout.PropertyField(size, sizeLabel);
169
170 serializedObject.ApplyModifiedProperties();
171 }
