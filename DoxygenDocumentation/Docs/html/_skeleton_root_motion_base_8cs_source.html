206 protected virtual void PhysicsUpdate (bool skeletonAnimationUsesFixedUpdate) {
207 Vector2 callbackDisplacement = tempSkeletonDisplacement;
208 float callbackRotation = tempSkeletonRotation;
209
210 bool isApplyAtRigidbodyAllowed = PhysicsUpdateRootMotionOverride == null || !disableOnOverride;
211 if (isApplyAtRigidbodyAllowed) {
212 if (rigidBody2D != null) {
213 Vector2 gravityAndVelocityMovement = Vector2.zero;
214 if (applyRigidbody2DGravity) {
215 float deltaTime = Time.fixedDeltaTime;
216 float deltaTimeSquared = (deltaTime * deltaTime);
217
218 rigidBody2D.velocity += rigidBody2D.gravityScale * Physics2D.gravity * deltaTime;
219 gravityAndVelocityMovement = 0.5f * rigidBody2D.gravityScale * Physics2D.gravity * deltaTimeSquared +
220 rigidBody2D.velocity * deltaTime;
221 }
222
223 Vector2 rigidbodyDisplacement2D = new Vector2(rigidbodyDisplacement.x, rigidbodyDisplacement.y);
224 // Note: MovePosition seems to be the only precise and reliable way to set movement delta,
225 // for both 2D and 3D rigidbodies.
226 // Setting velocity like "rigidBody2D.velocity = movement/deltaTime" works perfectly in mid-air
227 // without gravity and ground collision, unfortunately when on the ground, friction causes severe
228 // slowdown. Using a zero-friction PhysicsMaterial leads to sliding endlessly along the ground as
229 // soon as forces are applied. Additionally, there is no rigidBody2D.isGrounded, requiring our own
230 // checks.
231 rigidBody2D.MovePosition(gravityAndVelocityMovement + new Vector2(rigidBody2D.position.x, rigidBody2D.position.y)
232 + rigidbodyDisplacement2D + additionalRigidbody2DMovement);
233 rigidBody2D.MoveRotation(rigidbody2DRotation + rigidBody2D.rotation);
234 } else if (rigidBody != null) {
235 rigidBody.MovePosition(rigidBody.position
236 + new Vector3(rigidbodyDisplacement.x, rigidbodyDisplacement.y, rigidbodyDisplacement.z));
237 rigidBody.MoveRotation(rigidBody.rotation * rigidbodyLocalRotation);
238 }
239 }
240
241 previousRigidbodyRootMotion = rigidbodyDisplacement;
242 if (accumulatedUntilFixedUpdate) {
243 Vector2 parentBoneScale;
244 GetScaleAffectingRootMotion(out parentBoneScale);
245 ClearEffectiveBoneOffsets(parentBoneScale);
246 skeletonComponent.Skeleton.UpdateWorldTransform(Skeleton.Physics.Pose);
247 }
248 ClearRigidbodyTempMovement();
249
250 if (PhysicsUpdateRootMotionOverride != null)
251 PhysicsUpdateRootMotionOverride(this, callbackDisplacement, callbackRotation);
252 }

311 public void AdjustRootMotionToDistance (Vector2 distanceToTarget, int trackIndex = 0, bool adjustX = true, bool adjustY = true,
312 float minX = 0, float maxX = float.MaxValue, float minY = 0, float maxY = float.MaxValue,
313 bool allowXTranslation = false, bool allowYTranslation = false) {
314
315 Vector2 distanceToTargetSkeletonSpace = (Vector2)transform.InverseTransformVector(distanceToTarget);
316 Vector2 scaleAffectingRootMotion = GetScaleAffectingRootMotion();
317 if (UsesRigidbody)
318 distanceToTargetSkeletonSpace -= tempSkeletonDisplacement;
319
320 Vector2 remainingRootMotionSkeletonSpace = GetRemainingRootMotion(trackIndex);
321 remainingRootMotionSkeletonSpace.Scale(scaleAffectingRootMotion);
322 if (remainingRootMotionSkeletonSpace.x == 0)
323 remainingRootMotionSkeletonSpace.x = 0.0001f;
324 if (remainingRootMotionSkeletonSpace.y == 0)
325 remainingRootMotionSkeletonSpace.y = 0.0001f;
326
327 if (adjustX)
328 rootMotionScaleX = Math.Min(maxX, Math.Max(minX, distanceToTargetSkeletonSpace.x / remainingRootMotionSkeletonSpace.x));
329 if (adjustY)
330 rootMotionScaleY = Math.Min(maxY, Math.Max(minY, distanceToTargetSkeletonSpace.y / remainingRootMotionSkeletonSpace.y));
331
332 if (allowXTranslation)
333 rootMotionTranslateXPerY = (distanceToTargetSkeletonSpace.x - remainingRootMotionSkeletonSpace.x * rootMotionScaleX) / remainingRootMotionSkeletonSpace.y;
334 if (allowYTranslation)
335 rootMotionTranslateYPerX = (distanceToTargetSkeletonSpace.y - remainingRootMotionSkeletonSpace.y * rootMotionScaleY) / remainingRootMotionSkeletonSpace.x;
336 }

342 public Vector2 GetAnimationRootMotion (float startTime, float endTime,
343 Animation animation) {
344
345 if (startTime == endTime)
346 return Vector2.zero;
347
348 TranslateTimeline translateTimeline = animation.FindTranslateTimelineForBone(rootMotionBoneIndex);
349 TranslateXTimeline xTimeline = animation.FindTimelineForBone<TranslateXTimeline>(rootMotionBoneIndex);
350 TranslateYTimeline yTimeline = animation.FindTimelineForBone<TranslateYTimeline>(rootMotionBoneIndex);
351
352 // Non-looped base
353 Vector2 endPos = Vector2.zero;
354 Vector2 startPos = Vector2.zero;
355 if (translateTimeline != null) {
356 endPos = translateTimeline.Evaluate(endTime);
357 startPos = translateTimeline.Evaluate(startTime);
358 } else if (xTimeline != null || yTimeline != null) {
359 endPos = TimelineExtensions.Evaluate(xTimeline, yTimeline, endTime);
360 startPos = TimelineExtensions.Evaluate(xTimeline, yTimeline, startTime);
361 }
362 TransformConstraint[] transformConstraintsItems = skeletonComponent.Skeleton.TransformConstraints.Items;
363 foreach (int constraintIndex in this.transformConstraintIndices) {
364 TransformConstraint constraint = transformConstraintsItems[constraintIndex];
365 ApplyConstraintToPos(animation, constraint, constraintIndex, endTime, false, ref endPos);
366 ApplyConstraintToPos(animation, constraint, constraintIndex, startTime, true, ref startPos);
367 }
368 Vector2 currentDelta = endPos - startPos;
369
370 // Looped additions
371 if (startTime > endTime) {
372 Vector2 loopPos = Vector2.zero;
373 Vector2 zeroPos = Vector2.zero;
374 if (translateTimeline != null) {
375 loopPos = translateTimeline.Evaluate(animation.Duration);
376 zeroPos = translateTimeline.Evaluate(0);
377 } else if (xTimeline != null || yTimeline != null) {
378 loopPos = TimelineExtensions.Evaluate(xTimeline, yTimeline, animation.Duration);
379 zeroPos = TimelineExtensions.Evaluate(xTimeline, yTimeline, 0);
380 }
381 foreach (int constraintIndex in this.transformConstraintIndices) {
382 TransformConstraint constraint = transformConstraintsItems[constraintIndex];
383 ApplyConstraintToPos(animation, constraint, constraintIndex, animation.Duration, false, ref loopPos);
384 ApplyConstraintToPos(animation, constraint, constraintIndex, 0, false, ref zeroPos);
385 }
386 currentDelta += loopPos - zeroPos;
387 }
388 UpdateLastConstraintPos(transformConstraintsItems);
389 return currentDelta;
390 }

396 public float GetAnimationRootMotionRotation (float startTime, float endTime,
397 Animation animation) {
398
399 if (startTime == endTime)
400 return 0;
401
402 RotateTimeline rotateTimeline = animation.FindTimelineForBone<RotateTimeline>(rootMotionBoneIndex);
403
404 // Non-looped base
405 float endRotation = 0;
406 float startRotation = 0;
407 if (rotateTimeline != null) {
408 endRotation = rotateTimeline.Evaluate(endTime);
409 startRotation = rotateTimeline.Evaluate(startTime);
410 }
411 TransformConstraint[] transformConstraintsItems = skeletonComponent.Skeleton.TransformConstraints.Items;
412 foreach (int constraintIndex in this.transformConstraintIndices) {
413 TransformConstraint constraint = transformConstraintsItems[constraintIndex];
414 ApplyConstraintToRotation(animation, constraint, constraintIndex, endTime, false, ref endRotation);
415 ApplyConstraintToRotation(animation, constraint, constraintIndex, startTime, true, ref startRotation);
416 }
417 float currentDelta = endRotation - startRotation;
418
419 // Looped additions
420 if (startTime > endTime) {
421 float loopRotation = 0;
422 float zeroPos = 0;
423 if (rotateTimeline != null) {
424 loopRotation = rotateTimeline.Evaluate(animation.Duration);
425 zeroPos = rotateTimeline.Evaluate(0);
426 }
427 foreach (int constraintIndex in this.transformConstraintIndices) {
428 TransformConstraint constraint = transformConstraintsItems[constraintIndex];
429 ApplyConstraintToRotation(animation, constraint, constraintIndex, animation.Duration, false, ref loopRotation);
430 ApplyConstraintToRotation(animation, constraint, constraintIndex, 0, false, ref zeroPos);
431 }
432 currentDelta += loopRotation - zeroPos;
433 }
434 UpdateLastConstraintRotation(transformConstraintsItems);
435 return currentDelta;
436 }

438 void ApplyConstraintToPos (Animation animation, TransformConstraint constraint,
439 int constraintIndex, float time, bool useLastConstraintPos, ref Vector2 pos) {
440 TransformConstraintTimeline timeline = animation.FindTransformConstraintTimeline(constraintIndex);
441 if (timeline == null)
442 return;
443 Vector2 mixXY = timeline.EvaluateTranslateXYMix(time);
444 Vector2 invMixXY = timeline.EvaluateTranslateXYMix(time);
445 Vector2 constraintPos;
446 if (useLastConstraintPos)
447 constraintPos = transformConstraintLastPos[GetConstraintLastPosIndex(constraintIndex)];
448 else {
449 Bone targetBone = constraint.Target;
450 constraintPos = new Vector2(targetBone.X, targetBone.Y);
451 }
452 pos = new Vector2(
453 pos.x * invMixXY.x + constraintPos.x * mixXY.x,
454 pos.y * invMixXY.y + constraintPos.y * mixXY.y);
455 }
456
457 void ApplyConstraintToRotation (Animation animation, TransformConstraint constraint,
458 int constraintIndex, float time, bool useLastConstraintRotation, ref float rotation) {
459 TransformConstraintTimeline timeline = animation.FindTransformConstraintTimeline(constraintIndex);
460 if (timeline == null)
461 return;
462 float mixRotate = timeline.EvaluateRotateMix(time);
463 float invMixRotate = timeline.EvaluateRotateMix(time);
464 float constraintRotation;
465 if (useLastConstraintRotation)
466 constraintRotation = transformConstraintLastRotation[GetConstraintLastPosIndex(constraintIndex)];
467 else {
468 Bone targetBone = constraint.Target;
469 constraintRotation = targetBone.Rotation;
470 }
471 rotation = rotation * invMixRotate + constraintRotation * mixRotate;
472 }

476 TransformConstraint constraint = transformConstraintsItems[constraintIndex];
477 Bone targetBone = constraint.Target;

479 }

490 public RootMotionInfo GetAnimationRootMotionInfo (Animation animation, float currentTime) {
491 RootMotionInfo rootMotion = new RootMotionInfo();
492 float duration = animation.Duration;
493 float mid = duration * 0.5f;
494 rootMotion.timeIsPastMid = currentTime > mid;
495 TranslateTimeline timeline = animation.FindTranslateTimelineForBone(rootMotionBoneIndex);
496 if (timeline != null) {
497 rootMotion.start = timeline.Evaluate(0);
498 rootMotion.current = timeline.Evaluate(currentTime);
499 rootMotion.mid = timeline.Evaluate(mid);
500 rootMotion.end = timeline.Evaluate(duration);
501 return rootMotion;
502 }
503 TranslateXTimeline xTimeline = animation.FindTimelineForBone<TranslateXTimeline>(rootMotionBoneIndex);
504 TranslateYTimeline yTimeline = animation.FindTimelineForBone<TranslateYTimeline>(rootMotionBoneIndex);
505 if (xTimeline != null || yTimeline != null) {
506 rootMotion.start = TimelineExtensions.Evaluate(xTimeline, yTimeline, 0);
507 rootMotion.current = TimelineExtensions.Evaluate(xTimeline, yTimeline, currentTime);
508 rootMotion.mid = TimelineExtensions.Evaluate(xTimeline, yTimeline, mid);
509 rootMotion.end = TimelineExtensions.Evaluate(xTimeline, yTimeline, duration);
510 return rootMotion;
511 }
512 return rootMotion;
513 }

520 void FindTransformConstraintsAffectingBone () {
521 ExposedList<TransformConstraint> constraints = skeletonComponent.Skeleton.TransformConstraints;
522 TransformConstraint[] constraintsItems = constraints.Items;
523 for (int i = 0, n = constraints.Count; i < n; ++i) {
524 TransformConstraint constraint = constraintsItems[i];
525 if (constraint.Bones.Contains(rootMotionBone)) {
526 transformConstraintIndices.Add(i);
527 Bone targetBone = constraint.Target;
528 Vector2 constraintPos = new Vector2(targetBone.X, targetBone.Y);
529 transformConstraintLastPos.Add(constraintPos);
530 transformConstraintLastRotation.Add(targetBone.Rotation);
531 }
532 }

535 Vector2 GetTimelineMovementDelta (float startTime, float endTime,
536 TranslateXTimeline xTimeline, TranslateYTimeline yTimeline, Animation animation) {
537
538 Vector2 currentDelta;
539 if (startTime > endTime) // Looped
540 currentDelta =
541 (TimelineExtensions.Evaluate(xTimeline, yTimeline, animation.Duration)
542 - TimelineExtensions.Evaluate(xTimeline, yTimeline, startTime))
543 + (TimelineExtensions.Evaluate(xTimeline, yTimeline, endTime)
544 - TimelineExtensions.Evaluate(xTimeline, yTimeline, 0));
545 else if (startTime != endTime) // Non-looped
546 currentDelta = TimelineExtensions.Evaluate(xTimeline, yTimeline, endTime)
547 - TimelineExtensions.Evaluate(xTimeline, yTimeline, startTime);
548 else
549 currentDelta = Vector2.zero;
550 return currentDelta;
551 }

562 void HandleUpdateLocal (ISkeletonAnimation animatedSkeletonComponent) {
563 if (!this.isActiveAndEnabled)
564 return; // Root motion is only applied when component is enabled.
565
566 Vector2 boneLocalDelta = CalculateAnimationsMovementDelta();
567 Vector2 parentBoneScale;
568 Vector2 totalScale;
569 Vector2 skeletonTranslationDelta = GetSkeletonSpaceMovementDelta(boneLocalDelta, out parentBoneScale, out totalScale);
570 float skeletonRotationDelta = 0;
571 if (transformRotation) {
572 float boneLocalDeltaRotation = CalculateAnimationsRotationDelta();
573 boneLocalDeltaRotation *= rootMotionScaleRotation;
574 skeletonRotationDelta = GetSkeletonSpaceRotationDelta(boneLocalDeltaRotation, totalScale);
575 }
576
577 bool usesFixedUpdate = SkeletonAnimationUsesFixedUpdate;
578 ApplyRootMotion(skeletonTranslationDelta, skeletonRotationDelta, parentBoneScale, usesFixedUpdate);
579
580 if (usesFixedUpdate)
581 PhysicsUpdate(usesFixedUpdate);
582 }
583
584 void ApplyRootMotion (Vector2 skeletonTranslationDelta, float skeletonRotationDelta, Vector2 parentBoneScale,
585 bool skeletonAnimationUsesFixedUpdate) {
586
587 // Accumulated displacement is applied on the next Physics update in FixedUpdate.
588 // Until the next Physics update, tempSkeletonDisplacement and tempSkeletonRotation
589 // are offsetting bone locations to prevent stutter which would otherwise occur if
590 // we don't move every Update.
591 bool usesRigidbody = this.UsesRigidbody;
592 bool applyToTransform = !usesRigidbody && (ProcessRootMotionOverride == null || !disableOnOverride);
593 accumulatedUntilFixedUpdate = !applyToTransform && !skeletonAnimationUsesFixedUpdate;
594
595 if (ProcessRootMotionOverride != null)
596 ProcessRootMotionOverride(this, skeletonTranslationDelta, skeletonRotationDelta);
597
598 // Apply root motion to Transform or update values applied to RigidBody later (must happen in FixedUpdate).
599 if (usesRigidbody) {
600 rigidbodyDisplacement += transform.TransformVector(skeletonTranslationDelta);
601 if (skeletonRotationDelta != 0.0f) {
602 if (rigidBody != null) {
603 Quaternion addedWorldRotation = Quaternion.Euler(0, 0, skeletonRotationDelta);
604 rigidbodyLocalRotation = rigidbodyLocalRotation * addedWorldRotation;
605 } else if (rigidBody2D != null) {
606 Vector3 lossyScale = transform.lossyScale;
607 float rotationSign = lossyScale.x * lossyScale.y > 0 ? 1 : -1;
608 rigidbody2DRotation += rotationSign * skeletonRotationDelta;
609 }
610 }
611 } else if (applyToTransform) {
612 transform.position += transform.TransformVector(skeletonTranslationDelta);
613 if (skeletonRotationDelta != 0.0f) {
614 Vector3 lossyScale = transform.lossyScale;
615 float rotationSign = lossyScale.x * lossyScale.y > 0 ? 1 : -1;
616 transform.Rotate(0, 0, rotationSign * skeletonRotationDelta);
617 }
618 }
619
620 tempSkeletonDisplacement += skeletonTranslationDelta;
621 tempSkeletonRotation += skeletonRotationDelta;
622 if (accumulatedUntilFixedUpdate) {
623 SetEffectiveBoneOffsetsTo(tempSkeletonDisplacement, tempSkeletonRotation, parentBoneScale);
624 } else {
625 ClearEffectiveBoneOffsets(parentBoneScale);
626 }
627 }

648 Vector2 GetScaleAffectingRootMotion (out Vector2 parentBoneScale) {
649 Skeleton skeleton = skeletonComponent.Skeleton;
650 Vector2 totalScale = Vector2.one;
651 totalScale.x *= skeleton.ScaleX;
652 totalScale.y *= skeleton.ScaleY;
653
654 parentBoneScale = Vector2.one;
655 Bone scaleBone = rootMotionBone;
656 while ((scaleBone = scaleBone.Parent) != null) {
657#if USE_APPLIED_PARENT_SCALE
658 parentBoneScale.x *= scaleBone.AScaleX;
659 parentBoneScale.y *= scaleBone.AScaleY;
660#else
661 parentBoneScale.x *= scaleBone.ScaleX;
662 parentBoneScale.y *= scaleBone.ScaleY;
663#endif
664 }
665 totalScale = Vector2.Scale(totalScale, parentBoneScale);
666 totalScale *= AdditionalScale;
667 return totalScale;
668 }

670 Vector2 GetSkeletonSpaceMovementDelta (Vector2 boneLocalDelta, out Vector2 parentBoneScale, out Vector2 totalScale) {
671 Vector2 skeletonDelta = boneLocalDelta;
672 totalScale = GetScaleAffectingRootMotion(out parentBoneScale);
673 skeletonDelta.Scale(totalScale);
674
675 Vector2 rootMotionTranslation = new Vector2(
676 rootMotionTranslateXPerY * skeletonDelta.y,
677 rootMotionTranslateYPerX * skeletonDelta.x);
678

694 void SetEffectiveBoneOffsetsTo (Vector2 displacementSkeletonSpace, float rotationSkeletonSpace, Vector2 parentBoneScale) {
695
696 ApplyTransformConstraints();
697
698 // Move top level bones in opposite direction of the root motion bone
699 Skeleton skeleton = skeletonComponent.Skeleton;
700 foreach (Bone topLevelBone in topLevelBones) {
701 if (topLevelBone == rootMotionBone) {
702 if (transformPositionX) topLevelBone.X = displacementSkeletonSpace.x / skeleton.ScaleX;
703 if (transformPositionY) topLevelBone.Y = displacementSkeletonSpace.y / skeleton.ScaleY;
704 if (transformRotation) {
705 float rotationSign = skeleton.ScaleX * skeleton.ScaleY > 0 ? 1 : -1;
706 topLevelBone.Rotation = rotationSign * rotationSkeletonSpace;
707 }
708 } else {
709 bool useAppliedTransform = transformConstraintIndices.Count > 0;
710 float rootMotionBoneX = useAppliedTransform ? rootMotionBone.AX : rootMotionBone.X;
711 float rootMotionBoneY = useAppliedTransform ? rootMotionBone.AY : rootMotionBone.Y;
712
713 float offsetX = (initialOffset.x - rootMotionBoneX) * parentBoneScale.x;
714 float offsetY = (initialOffset.y - rootMotionBoneY) * parentBoneScale.y;
715
716 if (transformPositionX) topLevelBone.X = (displacementSkeletonSpace.x / skeleton.ScaleX) + offsetX;
717 if (transformPositionY) topLevelBone.Y = (displacementSkeletonSpace.y / skeleton.ScaleY) + offsetY;
718
719 if (transformRotation) {
720 float rootMotionBoneRotation = useAppliedTransform ? rootMotionBone.AppliedRotation : rootMotionBone.Rotation;
721
722 float parentBoneRotationSign = (parentBoneScale.x * parentBoneScale.y > 0 ? 1 : -1);
723 float offsetRotation = (initialOffsetRotation - rootMotionBoneRotation) * parentBoneRotationSign;
724
725 float skeletonRotationSign = skeleton.ScaleX * skeleton.ScaleY > 0 ? 1 : -1;
726 topLevelBone.Rotation = (rotationSkeletonSpace * skeletonRotationSign) + offsetRotation;
727 }
728 }
729 }
730 }
