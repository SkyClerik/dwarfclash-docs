97 public void RenderParts (ExposedList<SubmeshInstruction> instructions, int startSubmesh, int endSubmesh) {
98 LazyIntialize();
99
100 // STEP 1: Create instruction
101 MeshRendererBuffers.SmartMesh smartMesh = buffers.GetNextMesh();
102 currentInstructions.SetWithSubset(instructions, startSubmesh, endSubmesh);
103 bool updateTriangles = SkeletonRendererInstruction.GeometryNotEqual(currentInstructions, smartMesh.instructionUsed);
104
105 // STEP 2: Generate mesh buffers.
106 SubmeshInstruction[] currentInstructionsSubmeshesItems = currentInstructions.submeshInstructions.Items;
107 meshGenerator.Begin();
108 if (currentInstructions.hasActiveClipping) {
109 for (int i = 0; i < currentInstructions.submeshInstructions.Count; i++)
110 meshGenerator.AddSubmesh(currentInstructionsSubmeshesItems[i], updateTriangles);
111 } else {
112 meshGenerator.BuildMeshWithArrays(currentInstructions, updateTriangles);
113 }
114
115 buffers.UpdateSharedMaterials(currentInstructions.submeshInstructions);
116
117 // STEP 3: modify mesh.
118 Mesh mesh = smartMesh.mesh;
119
120 if (meshGenerator.VertexCount <= 0) { // Clear an empty mesh
121 updateTriangles = false;
122 mesh.Clear();
123 } else {
124 meshGenerator.FillVertexData(mesh);
125 if (updateTriangles) {
126 meshGenerator.FillTriangles(mesh);
127 meshRenderer.sharedMaterials = buffers.GetUpdatedSharedMaterialsArray();
128 } else if (buffers.MaterialsChangedInLastUpdate()) {
129 meshRenderer.sharedMaterials = buffers.GetUpdatedSharedMaterialsArray();
130 }
131 meshGenerator.FillLateVertexData(mesh);
132 }
133
134 meshFilter.sharedMesh = mesh;
135 smartMesh.instructionUsed.Set(currentInstructions);
136
137 if (OnMeshAndMaterialsUpdated != null)
138 OnMeshAndMaterialsUpdated(this);
139 }
